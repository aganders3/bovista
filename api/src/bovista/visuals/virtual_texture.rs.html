12/// Maximum number of physical atlas textures a single visual can hold.
13/// Visuals always pre-bind this many atlas textures in their shader bind
14/// group; visuals that request fewer get 1×1×1 dummy textures in the unused
15/// slots. The page-table encoding reserves 2 bits for `atlas_id`, so bumping
16/// this above 4 requires widening that field (see page_table.rs).
17pub const MAX_ATLAS_COUNT: usize = 4;

19/// Type alias for pending chunks queue (can be shared across threads)
20pub type PendingChunks = Arc<Mutex<HashMap<TileKey, TileData>>>;

22/// Set of tiles bovista currently wants in the atlas, keyed by full
23/// TileKey, value = priority (lower = more urgent). Rebuilt from
24/// scratch at the end of every prepare; loaders read it to decide
25/// what to fetch.
26pub type Wanted = Arc<Mutex<HashMap<TileKey, i32>>>;
27
28/// Snapshot of the wanted set as `(lod_level, t, z, y, x, priority)`
29/// tuples, sorted by priority (lower = more urgent; 0 = "user is looking
30/// at this t now", positive = prefetch offset). Shared by every language
31/// binding so the sort + projection lives in one place.
32pub fn wanted_sorted(wanted: &Wanted) -> Vec<(usize, u32, u32, u32, u32, i32)> {
33    let w = wanted.lock().unwrap();
34    let mut v: Vec<_> = w.iter()
35        .map(|(k, p)| (k.lod_level, k.t, k.z, k.y, k.x, *p))
36        .collect();
37    v.sort_by_key(|e| e.5);
38    v
39}

41/// Per-prepare timing + counts. Refreshed by every `prepare` /
42/// `prepare_volume` call, exposed via `VolumeVisual::stats()` /
43/// `Image::stats()` for performance diagnostics. Read it from
44/// the render loop after `scene.prepare` returns; it reflects the
45/// most recent prepare.
46#[derive(Default, Clone, Debug)]
47pub struct PrepareStats {
48    /// Frame number this snapshot covers.
49    pub frame: u64,
50    /// Tiles drained from `pending_chunks` and successfully written.
51    pub tiles_installed: usize,
52    /// Tiles drained from `pending_chunks` but dropped (slot_map
53    /// already had them, no longer visible, or `alloc_or_evict` returned
54    /// None because no resident slot was lower-priority).
55    pub tiles_dropped: usize,
56    /// Wallclock time spent in `process_pending_chunks`.
57    pub process_pending_ns: u64,
58    /// Wallclock time spent in `refresh_page_table`.
59    pub refresh_page_table_ns: u64,
60    /// Wallclock time spent in `publish_wanted`.
61    pub publish_wanted_ns: u64,
62    /// `queue.write_texture` calls into the atlas this prepare.
63    pub atlas_writes: usize,
64    /// `queue.write_texture` calls into the page table this prepare.
65    pub page_table_writes: usize,
66    /// `slot_map` size at end of prepare (atlas occupancy).
67    pub slot_map_len: usize,
68    /// `pending_chunks` size at end of prepare (tiles waiting after
69    /// any per-frame cap).
70    pub pending_len: usize,
71}

106/// Virtual-texture rendering strategy.
107///
108/// Maintains one or more physical 3D atlas textures plus a page table
109/// (2D-array texture) so the entire volume can be rendered with a single
110/// draw call. Up to `MAX_ATLAS_COUNT` atlases per visual — each atlas has
111/// its own free-list allocator and identical dimensions; the page-table
112/// entry encodes which atlas a tile lives in. atlas_count=1 produces the
113/// same on-GPU behavior as before, just with a 2-bit atlas_id field in
114/// the page table that's always 0.
115///
116/// Already reused across `Image` (slice plane) and `*Volume` (ray march)
117/// visuals — same atlas + page-table machinery, different consumer shader.
118/// TODO: make a single `VirtualTextureData` shareable across multiple
119/// visuals — 4-up orthogonal slices + a volume + cut planes all reading
120/// from one resident atlas, no duplicate decodes, single `wanted` set
121/// driven by the union of view frustums. Today each visual owns its VT
122/// exclusively; sharing needs the visuals to hold a handle (`Arc<RwLock>`
123/// or `Rc<RefCell>`) instead of the value, plus a `wanted` publisher
124/// that merges contributions from each visual's `prepare`.
125pub struct VirtualTextureData {
126    pub levels: Vec<LodLevelConfig>,
127
128    // Tile size assumed uniform across all LODs (validated in new()).
129    pub tile_size: (u32, u32, u32),  // (z, y, x)
130
131    // Physical atlas resources. `atlas_allocators.len() == atlas_count`;
132    // `atlas_textures` and `atlas_views` are always MAX_ATLAS_COUNT long —
133    // entries past `atlas_count` are 1×1×1 dummies (kept around so the
134    // shader bind group's 4-texture layout is always satisfied).
135    atlas_allocators: Vec<AtlasAllocator>,
136    pub atlas_textures: Vec<wgpu::Texture>,
137    pub atlas_views: Vec<wgpu::TextureView>,
138    pub atlas_count: usize,
139    pub atlas_cols: u32,
140    pub atlas_rows: u32,
141    pub atlas_layers: u32,
142    /// Total slot count across every real atlas. With a 4096-slot per-atlas
143    /// cap on the L40 and atlas_count=4, this can reach 16384. Used to bound
144    /// `publish_wanted` so we never publish more keys than the atlases can
145    /// physically hold — pending would otherwise grow unbounded at LOD0.
146    pub atlas_capacity: usize,
147
148    // Page table (owns its own texture + view)
149    pub page_table: PageTable,
150
151    // TileKey → (atlas_id, slot)
152    pub slot_map: HashMap<TileKey, (u32, u32)>,
153    // SPATIAL key (t forced to 0) → (atlas_id, slot) the page table currently
154    // points at for that spatial. Two uses, both about gating GPU writes:
155    //   1. `refresh_page_table` skips `queue.write_texture` calls for
156    //      spatials whose entry already points at the right slot.
157    //   2. On eviction, lets us detect when the slot a page-table entry
158    //      references is about to be recycled, so we can clear the
159    //      entry instead of letting the shader read garbage.
160    // NOT consulted by the eviction filter — that's the deadlock trap.
161    last_written_slot: HashMap<TileKey, (u32, u32)>,
162    // TileKey → last frame the tile was accessed
163    lru_map: HashMap<TileKey, u64>,
164
165    /// Tile bytes that have been fetched (by whoever the consumer
166    /// wired up) and are waiting to be uploaded to the atlas. Drained
167    /// at the top of every prepare via `process_pending_chunks`.
168    pub pending_chunks: PendingChunks,
169    /// Set of tiles bovista currently wants resident in the atlas,
170    /// keyed by full TileKey. Value is priority (lower = more urgent;
171    /// 0 means "user is looking at this t now"). Rebuilt atomically
172    /// from scratch at the end of every prepare; loaders read it and
173    /// decide what to fetch and in what order. No callbacks.
174    pub wanted: Wanted,
175
176    /// Diagnostic snapshot of the most recent prepare. Read by the
177    /// example's render loop to print `[perf]` lines.
178    pub stats: PrepareStats,
179
180    pub max_tiles: usize,
181    frame_counter: u64,
182
183    /// Visible tile spatial keys. All entries have `t = 0` — this set is
184    /// "which (lod, z, y, x) tiles are visible," independent of timepoint.
185    /// Pending-drain and LRU touching compare against the SPATIAL part of
186    /// stored slot_map keys.
187    pub visible_tile_keys: HashSet<TileKey>,
188    /// Latch so we warn at most once per under→over transition when the
189    /// visible working set can't fit in the atlas (guaranteed thrash),
190    /// instead of spamming every prepare.
191    over_capacity_warned: bool,
192    pub lod_bias: f32,
193    target_pixels_per_voxel: f32,
194    pub cached_ideal_lod: usize,
195    /// Timepoint the caller wants to display. Missing tiles at this t
196    /// fall back to coarser LODs via the shader's existing LOD-walk;
197    /// the shader's `desired_t` uniform compare also rejects any stale
198    /// entries left in the page table from a previous t. As fine tiles
199    /// arrive via `process_pending_chunks` the page table refines.
200    desired_t: u32,
201    /// Prefetch ±N timepoints around `desired_t` every prepare. Runs
202    /// through the SAME request path as current-t (slot_map check +
203    /// request_tile dedupe), so the atlas is the single source of
204    /// truth — already-resident tiles cost a HashMap lookup, never a
205    /// re-decode. 0 disables.
206    prefetch_lookahead: u32,
207    /// Total number of timepoints in the dataset. Used to clamp the
208    /// prefetch range; otherwise the strategy has no way to know it.
209    t_count: u32,
210}

212impl VirtualTextureData {
213    /// `max_tiles` is the *total* slot budget across all atlases; it's split
214    /// evenly into `atlas_count` per-atlas capacities. `atlas_count` is clamped
215    /// to `[1, MAX_ATLAS_COUNT]`.
216    pub fn new(
217        device: &Device,
218        lod_levels: Vec<LodLevelConfig>,
219        max_tiles: usize,
220        atlas_count: usize,
221    ) -> Self {
222        assert!(!lod_levels.is_empty(), "VirtualTextureData requires at least one LOD level");
223
224        let atlas_count = atlas_count.clamp(1, MAX_ATLAS_COUNT);
225        let tile_size = lod_levels[0].tile_size; // (z, y, x)
226
227        let (tile_d, tile_h, tile_w) = tile_size;
228
229        // Per-atlas slot budget. Round up so the sum >= max_tiles.
230        let capacity_per_atlas = max_tiles.div_ceil(atlas_count);
231
232        // Clamp the PER-ATLAS grid to the device's 3D texture dimension
233        // limits. Multi-atlas multiplies *capacity*, not per-texture size.
234        let max_dim = device.limits().max_texture_dimension_3d;
235        let max_cols   = (max_dim / tile_w).max(1);
236        let max_rows   = (max_dim / tile_h).max(1);
237        let max_layers = (max_dim / tile_d).max(1);
238        let effective_per_atlas =
239            capacity_per_atlas.min((max_cols * max_rows * max_layers) as usize);
240
241        // Distribute slots evenly across the three axes of each atlas.
242        let cbrt_cap = (effective_per_atlas as f64).cbrt().ceil() as u32;
243        let cols   = cbrt_cap.min(max_cols).max(1);
244        let rows   = cbrt_cap.min(max_rows).max(1);
245        let layers = (effective_per_atlas as u32).div_ceil(cols * rows)
246            .min(max_layers)
247            .max(1);
248
249        // cols*rows*layers may exceed effective_per_atlas due to grid rounding.
250        let per_atlas_capacity = effective_per_atlas.min((cols * rows * layers) as usize);
251        let atlas_capacity = per_atlas_capacity * atlas_count;
252        if atlas_capacity < max_tiles {
253            log::warn!(
254                "VT atlas clamped from {} to {} slots ({} × {}/atlas) by GPU \
255                 max_texture_dimension_3d={} (tile={}x{}x{} → max grid {}x{}x{}).",
256                max_tiles, atlas_capacity, atlas_count, per_atlas_capacity, max_dim,
257                tile_w, tile_h, tile_d,
258                max_cols, max_rows, max_layers,
259            );
260        }
261
262        // Create `atlas_count` real atlas textures + (MAX_ATLAS_COUNT - atlas_count)
263        // 1×1×1 dummies. Dummies satisfy the bind-group layout's fixed 4-texture
264        // arity; the shader never indexes them (atlas_id < atlas_count always).
265        let mut atlas_textures = Vec::with_capacity(MAX_ATLAS_COUNT);
266        let mut atlas_views = Vec::with_capacity(MAX_ATLAS_COUNT);
267        let mut atlas_allocators = Vec::with_capacity(atlas_count);
268        for i in 0..MAX_ATLAS_COUNT {
269            let is_real = i < atlas_count;
270            let (w, h, d) = if is_real {
271                (tile_w * cols, tile_h * rows, tile_d * layers)
272            } else {
273                (1, 1, 1)
274            };
275            let tex = device.create_texture(&wgpu::TextureDescriptor {
276                label: Some(if is_real { "VT Atlas" } else { "VT Atlas (dummy)" }),
277                size: wgpu::Extent3d { width: w, height: h, depth_or_array_layers: d },
278                mip_level_count: 1,
279                sample_count: 1,
280                dimension: wgpu::TextureDimension::D3,
281                format: wgpu::TextureFormat::R16Float,
282                usage: wgpu::TextureUsages::TEXTURE_BINDING | wgpu::TextureUsages::COPY_DST,
283                view_formats: &[],
284            });
285            let view = tex.create_view(&wgpu::TextureViewDescriptor::default());
286            atlas_textures.push(tex);
287            atlas_views.push(view);
288            if is_real {
289                atlas_allocators.push(AtlasAllocator::new(per_atlas_capacity));
290            }
291        }
292
293        let page_table = PageTable::new(device, &lod_levels);
294
295        Self {
296            tile_size,
297            atlas_allocators,
298            atlas_textures,
299            atlas_views,
300            atlas_count,
301            atlas_cols: cols,
302            atlas_rows: rows,
303            atlas_layers: layers,
304            atlas_capacity,
305            page_table,
306            slot_map: HashMap::new(),
307            last_written_slot: HashMap::new(),
308            lru_map: HashMap::new(),
309            pending_chunks: Arc::new(Mutex::new(HashMap::new())),
310            wanted: Arc::new(Mutex::new(HashMap::new())),
311            stats: PrepareStats::default(),
312            max_tiles,
313            frame_counter: 0,
314            visible_tile_keys: HashSet::new(),
315            over_capacity_warned: false,
316            lod_bias: 0.0,
317            target_pixels_per_voxel: 1.0,
318            cached_ideal_lod: 0,
319            levels: lod_levels,
320            desired_t: 0,
321            prefetch_lookahead: 0,
322            t_count: 1,
323        }
324    }

343    pub fn desired_t(&self) -> u32 { self.desired_t }

345    /// (resident, visible) — how many of the visible spatial tiles at
346    /// `desired_t` are actually in the atlas right now. Useful for
347    /// progress reporting; returns (0, 0) before the first prepare.
348    pub fn current_t_load_status(&self) -> (usize, usize) {
349        let target = self.desired_t;
350        let resident = self.visible_tile_keys.iter()
351            .filter(|s| {
352                let key = TileKey { lod_level: s.lod_level, t: target,
353                                    z: s.z, y: s.y, x: s.x };
354                self.slot_map.contains_key(&key)
355            })
356            .count();
357        (resident, self.visible_tile_keys.len())
358    }

362    /// Drop all resident tiles. Now that `TileKey` includes `t`, the
363    /// preferred way to switch timepoints is `set_desired_timepoint(t)` —
364    /// adjacent t's stay resident in the atlas as cache and the page table
365    /// flips atomically once the new t is fully loaded (no flicker).
366    /// `clear_atlas` is retained for unusual cases where the underlying
367    /// data shape itself has changed.
368    pub fn clear_atlas(&mut self, queue: &Queue) {
369        // Clear every page-table entry that's been written.
370        for spatial in self.last_written_slot.keys() {
371            self.page_table.clear(queue, spatial.lod_level, spatial.z, spatial.y, spatial.x);
372        }
373        self.last_written_slot.clear();
374        self.slot_map.clear();
375        self.lru_map.clear();
376        self.wanted.lock().unwrap().clear();
377        self.pending_chunks.lock().unwrap().clear();
378        for alloc in &mut self.atlas_allocators {
379            alloc.reset();
380        }
381    }

383    fn process_pending_chunks(&mut self, queue: &Queue) {
384        let started = web_time::Instant::now();
385        let pending: Vec<(TileKey, TileData)> = {
386            let mut guard = self.pending_chunks.lock().unwrap();
387            guard.drain().collect()
388        };
389        if pending.is_empty() {
390            self.stats.process_pending_ns = started.elapsed().as_nanos() as u64;
391            return;
392        }
393
394        // Cap how many tiles we install per frame. Each
395        // queue.write_texture on the GL backend is ~3 ms; 16 tiles is
396        // ~48 ms — enough headroom inside a 33 ms frame budget for
397        // render-thread work to stay responsive. Excess pending tiles
398        // get pushed back into `pending` (HashMap, no dedup needed
399        // since publish_wanted skips pending entries so workers won't
400        // re-fetch them).
401        const MAX_TILES_PER_FRAME: usize = 16;
402
403        // Build the eviction candidate list ONCE
404        // (single O(N log N) sort vs O(N) scan per tile).
405        let current_t = self.desired_t;
406        let frame_counter = self.frame_counter;
407        let mut candidates: Vec<(u64, u64, TileKey, (u32, u32))> = self.slot_map.iter()
408            .map(|(k, &id_slot)| {
409                let eff = if self.visible_tile_keys.contains(&k.spatial()) {
410                    (k.t as i64 - current_t as i64).unsigned_abs()
411                } else {
412                    u64::MAX
413                };
414                let age = frame_counter
415                    .saturating_sub(*self.lru_map.get(k).unwrap_or(&0));
416                (eff, age, *k, id_slot)
417            })
418            .collect();
419        candidates.sort_by_key(|&(eff, age, _, _)| (eff, age));
420
421        // Sort pending by install priority so current-t visible tiles
422        // install before prefetch-t tiles within the per-frame budget.
423        // Without this, `pending.drain()` returns tiles in HashMap order
424        // (effectively random) — and when pending is dominated by
425        // prefetch entries, current-t tiles starve and `loaded N/M`
426        // stays well below M even though the missing tiles are sitting
427        // in pending.
428        let mut pending = pending;
429        pending.sort_by_key(|(k, _)| {
430            let eff = if self.visible_tile_keys.contains(&k.spatial()) {
431                (k.t as i64 - current_t as i64).unsigned_abs()
432            } else {
433                u64::MAX
434            };
435            (eff, k.lod_level as u64)
436        });
437
438        let mut installed = 0usize;
439        let mut dropped = 0usize;
440        let mut page_table_writes = 0usize;
441        let mut deferred: Vec<(TileKey, TileData)> = Vec::new();
442
443        for (key, data) in pending {
444            if installed >= MAX_TILES_PER_FRAME {
445                // Frame budget spent — defer the rest. Lower-priority
446                // tiles fall to the back of next frame's sort.
447                deferred.push((key, data));
448                continue;
449            }
450            if self.slot_map.contains_key(&key) {
451                dropped += 1;
452                continue;
453            }
454            if !self.visible_tile_keys.contains(&key.spatial()) {
455                dropped += 1;
456                continue;
457            }
458            let Some((atlas_id, slot)) = self.alloc_or_evict(queue, key.t, &mut candidates) else {
459                dropped += 1;
460                continue;
461            };
462
463            self.write_tile_to_atlas(queue, atlas_id, slot, &data);
464
465            let TileKey { lod_level, t, z, y, x } = key;
466            if t == self.desired_t {
467                self.page_table.update(queue, lod_level, z, y, x, atlas_id, t, slot);
468                self.last_written_slot.insert(key.spatial(), (atlas_id, slot));
469                page_table_writes += 1;
470            }
471
472            self.slot_map.insert(key, (atlas_id, slot));
473            self.lru_map.insert(key, self.frame_counter);
474            installed += 1;
475        }
476        // Re-queue deferred entries into pending for next frame.
477        if !deferred.is_empty() {
478            let mut guard = self.pending_chunks.lock().unwrap();
479            for (key, data) in deferred {
480                guard.insert(key, data);
481            }
482        }
483        self.stats.process_pending_ns = started.elapsed().as_nanos() as u64;
484        self.stats.tiles_installed = installed;
485        self.stats.tiles_dropped = dropped;
486        self.stats.atlas_writes = installed;
487        self.stats.page_table_writes = page_table_writes;
488    }

490    fn write_tile_to_atlas(&self, queue: &Queue, atlas_id: u32, slot: u32, data: &TileData) {
491        let col   = slot % self.atlas_cols;
492        let row   = (slot / self.atlas_cols) % self.atlas_rows;
493        let layer = slot / (self.atlas_cols * self.atlas_rows);
494        let (tile_d, tile_h, tile_w) = self.tile_size;
495        let origin = wgpu::Origin3d {
496            x: col   * tile_w,
497            y: row   * tile_h,
498            z: layer * tile_d,
499        };
500        let atlas_texture = &self.atlas_textures[atlas_id as usize];
501        let bpv = data.bytes_per_voxel() as usize;
502
503        // Defensive: reject a tile we can't safely upload, rather than panic.
504        // A bad shape can come from a caller that supplies the shape separately
505        // from the data (the WASM binding) — e.g. transposed axes. Two ways it
506        // bites, both of which would otherwise panic on an out-of-range slice
507        // (and in WASM a panic aborts the module and poisons the viewer for
508        // every later call):
509        //   1. byte count != z*y*x  → the source row read below overruns;
510        //   2. any extent > the slot → the padded *destination* row write
511        //      overruns, even when the byte count happens to match (e.g. axes
512        //      transposed to the same total). Boundary tiles are always <= the
513        //      slot, so a larger extent is by definition malformed.
514        // Native/Python derive the shape from the array and never hit either.
515        let expected = data.z_shape as usize
516            * data.y_shape as usize
517            * data.x_shape as usize
518            * bpv;
519        if data.data.len() != expected
520            || data.x_shape > tile_w
521            || data.y_shape > tile_h
522            || data.z_shape > tile_d
523        {
524            log::warn!(
525                "write_tile_to_atlas: skipping malformed tile — {} bytes, shape \
526                 z={} y={} x={} (expected {expected} bytes, slot {tile_d}x{tile_h}x{tile_w})",
527                data.data.len(), data.z_shape, data.y_shape, data.x_shape,
528            );
529            return;
530        }
531
532        // Boundary tiles (z/y/x edge of the volume at this LOD) have
533        // smaller extent than the slot. Without zero-padding, the
534        // remainder of the slot would still hold whatever a previous
535        // tile left there, and the shader's per-LOD data_scale would
536        // sample into that stale data.
537        let is_full =
538            data.x_shape == tile_w && data.y_shape == tile_h && data.z_shape == tile_d;
539        if is_full {
540            queue.write_texture(
541                wgpu::TexelCopyTextureInfo {
542                    texture: atlas_texture,
543                    mip_level: 0,
544                    origin,
545                    aspect: wgpu::TextureAspect::All,
546                },
547                &data.data,
548                wgpu::TexelCopyBufferLayout {
549                    offset: 0,
550                    bytes_per_row: Some(data.x_shape * bpv as u32),
551                    rows_per_image: Some(data.y_shape),
552                },
553                wgpu::Extent3d {
554                    width: data.x_shape,
555                    height: data.y_shape,
556                    depth_or_array_layers: data.z_shape,
557                },
558            );
559        } else {
560            let stride_row_dst = tile_w as usize * bpv;
561            let stride_slice_dst = stride_row_dst * tile_h as usize;
562            let stride_row_src = data.x_shape as usize * bpv;
563            let mut padded = vec![0u8; stride_slice_dst * tile_d as usize];
564            for z in 0..data.z_shape as usize {
565                for y in 0..data.y_shape as usize {
566                    let dst = z * stride_slice_dst + y * stride_row_dst;
567                    let src = z * (data.y_shape as usize) * stride_row_src
568                            + y * stride_row_src;
569                    padded[dst..dst + stride_row_src]
570                        .copy_from_slice(&data.data[src..src + stride_row_src]);
571                }
572            }
573            queue.write_texture(
574                wgpu::TexelCopyTextureInfo {
575                    texture: atlas_texture,
576                    mip_level: 0,
577                    origin,
578                    aspect: wgpu::TextureAspect::All,
579                },
580                &padded,
581                wgpu::TexelCopyBufferLayout {
582                    offset: 0,
583                    bytes_per_row: Some(tile_w * bpv as u32),
584                    rows_per_image: Some(tile_h),
585                },
586                wgpu::Extent3d {
587                    width: tile_w,
588                    height: tile_h,
589                    depth_or_array_layers: tile_d,
590                },
591            );
592        }
593    }

595    /// For every visible spatial, point the page table at the slot
596    /// holding the `desired_t` tile if resident. Tiles at other
597    /// timepoints sit in slot_map silently — the shader's `desired_t`
598    /// compare rejects any page-table entries left at other t's, so no
599    /// explicit clear is needed when desired_t changes.
600    ///
601    /// Two cheap gates keep the GL traffic bounded:
602    ///   - skip spatials with no resident `desired_t` tile (slot_map miss),
603    ///   - skip spatials where `last_written_slot` already shows this slot.
604    ///
605    /// In steady state (page table fully wired for desired_t), this runs
606    /// zero `queue.write_texture` calls.
607    ///
608    /// Rebuild `wanted` atomically from current state: every visible
609    /// spatial at `desired_t` (priority 0), plus the next N prefetch
610    /// timepoints (priority = offset), skipping anything already in
611    /// `slot_map`. Loaders read this map and decide what to fetch and
612    /// in what order.
613    fn publish_wanted(&mut self) {
614        let t0 = web_time::Instant::now();
615        let lookahead = self.prefetch_lookahead as i32;
616
617        // If the visible working set alone exceeds the atlas, not every
618        // visible tile can be resident at once — they evict each other and
619        // get re-requested every frame (the "loaded plateaus below visible,
620        // pending churns" thrash). Warn once per under→over transition so
621        // the caller knows to raise max_tiles / atlas_count / VRAM budget.
622        let visible = self.visible_tile_keys.len();
623        if visible > self.atlas_capacity {
624            if !self.over_capacity_warned {
625                log::warn!(
626                    "VT thrash: {visible} tiles visible but atlas holds only {} \
627                     (max_tiles={}). Visible tiles can't all stay resident — raise \
628                     max_tiles / atlas_count (or the VRAM budget) above {visible}.",
629                    self.atlas_capacity, self.max_tiles,
630                );
631                self.over_capacity_warned = true;
632            }
633        } else {
634            self.over_capacity_warned = false;
635        }
636        // Snapshot pending keys so a worker doesn't re-fetch a tile
637        // that's already decoded and waiting in line to install.
638        let pending_keys: HashSet<TileKey> = {
639            let guard = self.pending_chunks.lock().unwrap();
640            guard.keys().copied().collect()
641        };
642
643        // Collect candidates with priority = offset (0 = current t).
644        // We may build more than `atlas_capacity`; we sort + truncate
645        // below so the published set fits.
646        let mut candidates: Vec<(TileKey, i32)> = Vec::with_capacity(
647            self.visible_tile_keys.len() * (1 + lookahead.max(0) as usize),
648        );
649        for spatial in &self.visible_tile_keys {
650            for offset in 0..=lookahead {
651                let t_i = self.desired_t as i32 + offset;
652                if t_i < 0 || (self.t_count > 1 && t_i >= self.t_count as i32) {
653                    continue;
654                }
655                let key = TileKey {
656                    lod_level: spatial.lod_level,
657                    t: t_i as u32,
658                    z: spatial.z, y: spatial.y, x: spatial.x,
659                };
660                if self.slot_map.contains_key(&key) { continue; }
661                if pending_keys.contains(&key) { continue; }
662                candidates.push((key, offset));
663            }
664        }
665
666        // Bound the wanted set so pending doesn't accumulate to host
667        // RAM OOM. The earlier `atlas_capacity - pending` cap kept
668        // pending bounded but it plateaued near atlas_capacity (~4 GB+
669        // host RAM at 4 MB/tile) because the rate equilibrium settled
670        // wherever publish lets enough through to match the install
671        // drain rate. We want pending to plateau MUCH lower.
672        //
673        // PENDING_TARGET is a small constant: workers can only race
674        // ahead of the installer by this many tiles. Equilibrium
675        // pending sits at ~PENDING_TARGET minus the few keys in flight
676        // at any moment. Tradeoff: prefetch lookahead at LOD0 (where
677        // visible count dominates) is silently bounded by this target,
678        // because the same wanted slots prioritize current-t first.
679        //
680        // 512 ≈ 2 GB host RAM at 4 MB/tile, well under any sane
681        // budget. The renderer drains 16/frame so 512 is ~1 s of
682        // install buffer.
683        const PENDING_TARGET: usize = 512;
684        let effective_cap = PENDING_TARGET.saturating_sub(pending_keys.len());
685        if candidates.len() > effective_cap {
686            // Sort by (priority, lod_level desc) so we keep current-t
687            // tiles first, then nearest-t prefetch; among same-priority
688            // ties, prefer coarser LOD (which blocks in faster).
689            candidates.sort_by_key(|&(k, p)| (p, -(k.lod_level as i32)));
690            candidates.truncate(effective_cap);
691        }
692
693        let mut next: HashMap<TileKey, i32> = HashMap::with_capacity(candidates.len());
694        for (key, prio) in candidates {
695            next.insert(key, prio);
696        }
697        *self.wanted.lock().unwrap() = next;
698        self.stats.publish_wanted_ns = t0.elapsed().as_nanos() as u64;
699    }

701    /// O(visible) per call; cheap enough to run every prepare.
702    fn refresh_page_table(&mut self, queue: &Queue) {
703        let t0 = web_time::Instant::now();
704        let target = self.desired_t;
705        let mut writes = 0usize;
706        let visible: Vec<TileKey> =
707            self.visible_tile_keys.iter().copied().collect();
708        for spatial in &visible {
709            let key = TileKey { lod_level: spatial.lod_level, t: target,
710                                z: spatial.z, y: spatial.y, x: spatial.x };
711            let Some(&(atlas_id, slot)) = self.slot_map.get(&key) else { continue; };
712            if self.last_written_slot.get(spatial) == Some(&(atlas_id, slot)) { continue; }
713            self.page_table.update(queue,
714                spatial.lod_level, spatial.z, spatial.y, spatial.x, atlas_id, target, slot);
715            self.last_written_slot.insert(*spatial, (atlas_id, slot));
716            writes += 1;
717        }
718        self.stats.refresh_page_table_ns = t0.elapsed().as_nanos() as u64;
719        // Add to atlas-vs-page-table breakdown so it's clear how many
720        // page-table writes came from this path vs process_pending.
721        self.stats.page_table_writes += writes;
722    }

724    /// Return an atlas slot suitable for storing a tile at `tile_t`. If the
725    /// allocator has a free slot, use it. Otherwise pick the worst-priority
726    /// resident slot and evict it — but only if it really is lower-priority
727    /// than the incoming tile.
728    ///
729    /// Effective distance per slot:
730    ///   non-visible spatial → u64::MAX (always evictable)
731    ///   visible spatial     → |slot.t - desired_t|
732    /// Within a tier, oldest LRU loses. The incoming tile claims the slot
733    /// iff `eff > |tile_t - desired_t|`, so a current-t arrival (dist 0)
734    /// can claim any non-current slot; a prefetch at distance D can only
735    /// claim slots at distance > D.
736    /// Return a slot for a tile at `tile_t`. Free-list first; if
737    /// exhausted, evict the worst-priority entry from `candidates`
738    /// (a sorted-ascending list built once per `process_pending`
739    /// batch — see there for the priority rule).
740    fn alloc_or_evict(
741        &mut self,
742        queue: &Queue,
743        tile_t: u32,
744        candidates: &mut Vec<(u64, u64, TileKey, (u32, u32))>,
745    ) -> Option<(u32, u32)> {
746        // Free-list first: walk atlases in order and return the first slot.
747        for (id, alloc) in self.atlas_allocators.iter_mut().enumerate() {
748            if let Some(slot) = alloc.alloc() {
749                return Some((id as u32, slot));
750            }
751        }
752        let current = self.desired_t;
753        let incoming_dist = (tile_t as i64 - current as i64).unsigned_abs();
754        // Pop from the prebuilt sorted list (tail is most-evictable).
755        // Skip entries whose slot already changed hands earlier in
756        // this same batch (e.g. another iteration already evicted
757        // them).
758        while let Some((eff, age, vkey, (vatlas, vslot))) = candidates.pop() {
759            if eff <= incoming_dist {
760                // Push back: a later iteration with smaller
761                // incoming_dist may still be able to claim this.
762                candidates.push((eff, age, vkey, (vatlas, vslot)));
763                return None;
764            }
765            if self.slot_map.get(&vkey) != Some(&(vatlas, vslot)) {
766                continue;
767            }
768            self.slot_map.remove(&vkey);
769            self.lru_map.remove(&vkey);
770            self.atlas_allocators[vatlas as usize].free(vslot);
771            let spatial = vkey.spatial();
772            if self.last_written_slot.get(&spatial) == Some(&(vatlas, vslot)) {
773                let TileKey { lod_level, z, y, x, .. } = vkey;
774                self.page_table.clear(queue, lod_level, z, y, x);
775                self.last_written_slot.remove(&spatial);
776            }
777            // The freed slot is in the same atlas we just freed from —
778            // grab it directly rather than re-walking the allocator list.
779            return self.atlas_allocators[vatlas as usize]
780                .alloc()
781                .map(|s| (vatlas, s));
782        }
783        None
784    }

788    fn select_lod_for_tile(
789        &self,
790        tile_center: Vec3,
791        tile_world_size: f32,
792        camera_info: &crate::visual::CameraInfo,
793    ) -> usize {
794        let screen_pixels = match camera_info.projection_mode {
795            crate::ProjectionMode::Perspective => {
796                let distance = (tile_center - camera_info.position).length();
797                let min_distance = tile_world_size * 0.1;
798                if distance < min_distance {
799                    return 0;
800                }
801                let focal_length = camera_info.viewport_height as f32
802                    / (2.0 * (camera_info.fov_y / 2.0).tan());
803                (tile_world_size / distance) * focal_length
804            }
805            crate::ProjectionMode::Orthographic => {
806                (tile_world_size / camera_info.ortho_height) * camera_info.viewport_height as f32
807            }
808        };
809
810        let lod0 = &self.levels[0];
811        let voxels_per_tile = lod0.tile_size.2 as f32;
812        let pixels_per_voxel = screen_pixels / voxels_per_tile;
813        let target = self.target_pixels_per_voxel * 2.0_f32.powf(-self.lod_bias);
814
815        for (lod, level) in self.levels.iter().enumerate() {
816            if pixels_per_voxel * level.scale_factor >= target {
817                return lod;
818            }
819        }
820        self.levels.len() - 1
821    }

823    fn compute_tile_aabb(&self, lod_level: usize, tile_pos: (u32, u32, u32)) -> AABB {
824        let level = &self.levels[lod_level];
825        let (tz, ty, tx) = tile_pos;
826        let (tile_size_z, tile_size_y, tile_size_x) = level.tile_size;
827        let (voxel_z, voxel_y, voxel_x) = level.voxel_size;
828        let (trans_z, trans_y, trans_x) = level.translation;
829
830        let min = Vec3::new(
831            tx as f32 * tile_size_x as f32 * voxel_x + trans_x,
832            ty as f32 * tile_size_y as f32 * voxel_y + trans_y,
833            tz as f32 * tile_size_z as f32 * voxel_z + trans_z,
834        );
835
836        let (vol_z, vol_y, vol_x) = level.volume_size;
837        let actual_size_x = tile_size_x.min(vol_x - tx * tile_size_x);
838        let actual_size_y = tile_size_y.min(vol_y - ty * tile_size_y);
839        let actual_size_z = tile_size_z.min(vol_z - tz * tile_size_z);
840
841        let max = min + Vec3::new(
842            actual_size_x as f32 * voxel_x,
843            actual_size_y as f32 * voxel_y,
844            actual_size_z as f32 * voxel_z,
845        );
846        AABB::new(min, max)
847    }

849    fn plane_aabb_intersection(plane: &SlicePlane, aabb: &AABB) -> bool {
850        let plane_pos = Vec3::from(plane.position);
851        let plane_normal = Vec3::from(plane.normal);
852
853        let corners = [
854            aabb.min,
855            Vec3::new(aabb.max.x, aabb.min.y, aabb.min.z),
856            Vec3::new(aabb.min.x, aabb.max.y, aabb.min.z),
857            Vec3::new(aabb.max.x, aabb.max.y, aabb.min.z),
858            Vec3::new(aabb.min.x, aabb.min.y, aabb.max.z),
859            Vec3::new(aabb.max.x, aabb.min.y, aabb.max.z),
860            Vec3::new(aabb.min.x, aabb.max.y, aabb.max.z),
861            aabb.max,
862        ];
863
864        let mut min_dist = f32::MAX;
865        let mut max_dist = f32::MIN;
866        for corner in &corners {
867            let dist = plane_normal.dot(*corner - plane_pos);
868            min_dist = min_dist.min(dist);
869            max_dist = max_dist.max(dist);
870        }
871
872        let aabb_size = (aabb.max - aabb.min).length();
873        let epsilon = aabb_size * 0.001;
874        min_dist <= epsilon && max_dist >= -epsilon
875    }

877    fn update_visible_tiles(
878        &mut self,
879        slice_plane: &SlicePlane,
880        camera_info: &crate::visual::CameraInfo,
881    ) {
882        self.visible_tile_keys.clear();
883
884        let lod0 = &self.levels[0];
885        let typical_tile_size = lod0.tile_size.2 as f32 * lod0.voxel_size.2;
886
887        let plane_normal = Vec3::from_array(slice_plane.normal);
888        let plane_point = Vec3::from_array(slice_plane.position);
889        // Signed distance: positive = camera on the normal-facing side.
890        let signed_dist = (camera_info.position - plane_point).dot(plane_normal);
891
892        // Project the camera position onto the slice plane.
893        let center_point = camera_info.position - plane_normal * signed_dist;
894        let ideal_lod = self.select_lod_for_tile(center_point, typical_tile_size, camera_info);
895        self.cached_ideal_lod = ideal_lod;
896
897        // Hierarchical traversal from a manageable coarse LOD down to ideal_lod.
898        let mut start_lod = ideal_lod;
899        for lod in ideal_lod..self.levels.len() {
900            let level = &self.levels[lod];
901            let grid = level.grid_size();
902            if (grid.0 as usize) * (grid.1 as usize) * (grid.2 as usize) <= 1000 {
903                start_lod = lod;
904                break;
905            }
906        }
907
908        let start_level = &self.levels[start_lod];
909        let start_grid = start_level.grid_size();
910        let mut active_regions: Vec<(usize, u32, u32, u32)> = Vec::new();
911
912        for tz in 0..start_grid.0 {
913            for ty in 0..start_grid.1 {
914                for tx in 0..start_grid.2 {
915                    let aabb = self.compute_tile_aabb(start_lod, (tz, ty, tx));
916                    let margin = (aabb.max - aabb.min) * 0.05;
917                    if !camera_info.frustum.contains_aabb_xy(aabb.min - margin, aabb.max + margin) {
918                        continue;
919                    }
920                    if Self::plane_aabb_intersection(slice_plane, &aabb) {
921                        active_regions.push((start_lod, tz, ty, tx));
922                    }
923                }
924            }
925        }
926
927        // Add start_lod tiles as fallback coverage.  Only start_lod and ideal_lod
928        // go into visible_tile_keys — adding every intermediate LOD was inflating the
929        // visible count far above max_tiles and causing constant eviction/reload churn.
930        for &(lod, tz, ty, tx) in &active_regions {
931            self.visible_tile_keys.insert(TileKey::new(lod, 0, tz, ty, tx));
932        }
933
934        for current_lod in (ideal_lod..start_lod).rev() {
935            let mut next_regions = Vec::new();
936            for (parent_lod, pz, py, px) in &active_regions {
937                if *parent_lod != current_lod + 1 {
938                    next_regions.push((*parent_lod, *pz, *py, *px));
939                    continue;
940                }
941                let parent_grid_dims = self.levels[*parent_lod].grid_size();
942                let child_grid = self.levels[current_lod].grid_size();
943                // Children-per-parent per axis: derived from grid sizes so floating-point
944                // scale_factor imprecision can't cause tiles to be silently skipped.
945                let cpp_z = child_grid.0.div_ceil(parent_grid_dims.0).max(1);
946                let cpp_y = child_grid.1.div_ceil(parent_grid_dims.1).max(1);
947                let cpp_x = child_grid.2.div_ceil(parent_grid_dims.2).max(1);
948
949                for dz in 0..cpp_z {
950                    for dy in 0..cpp_y {
951                        for dx in 0..cpp_x {
952                            let (cz, cy, cx) = (pz * cpp_z + dz, py * cpp_y + dy, px * cpp_x + dx);
953                            if cz >= child_grid.0 || cy >= child_grid.1 || cx >= child_grid.2 {
954                                continue;
955                            }
956                            let aabb = self.compute_tile_aabb(current_lod, (cz, cy, cx));
957                            let margin = (aabb.max - aabb.min) * 0.05;
958                            if !camera_info.frustum.contains_aabb_xy(aabb.min - margin, aabb.max + margin) {
959                                continue;
960                            }
961                            if Self::plane_aabb_intersection(slice_plane, &aabb) {
962                                next_regions.push((current_lod, cz, cy, cx));
963                            }
964                        }
965                    }
966                }
967            }
968            active_regions = next_regions;
969        }
970
971        // Add the ideal_lod tiles — these are the ones we actually want to display.
972        for &(lod, tz, ty, tx) in &active_regions {
973            self.visible_tile_keys.insert(TileKey::new(lod, 0, tz, ty, tx));
974        }
975    }

977    fn update_visible_tiles_volume(
978        &mut self,
979        camera_info: &crate::visual::CameraInfo,
980    ) {
981        self.visible_tile_keys.clear();
982        if self.levels.is_empty() { return; }
983
984        // Seed the stack from the coarsest (single or few tile) LOD.
985        let coarsest = self.levels.len() - 1;
986        let coarse_grid = self.levels[coarsest].grid_size();
987        let mut stack: Vec<(usize, u32, u32, u32)> = Vec::with_capacity(
988            (coarse_grid.0 * coarse_grid.1 * coarse_grid.2) as usize,
989        );
990        for tz in 0..coarse_grid.0 {
991            for ty in 0..coarse_grid.1 {
992                for tx in 0..coarse_grid.2 {
993                    stack.push((coarsest, tz, ty, tx));
994                }
995            }
996        }
997
998        // cached_ideal_lod from the volume centre — used as the shader's target_lod starting
999        // point so the page-table walk is efficient on average.
1000        {
1001            let lod0 = &self.levels[0];
1002            let (vz, vy, vx) = lod0.volume_size;
1003            let (sz, sy, sx) = lod0.voxel_size;
1004            let (tz0, ty0, tx0) = lod0.translation;
1005            let vol_center = Vec3::new(
1006                tx0 + vx as f32 * sx * 0.5,
1007                ty0 + vy as f32 * sy * 0.5,
1008                tz0 + vz as f32 * sz * 0.5,
1009            );
1010            let typical = lod0.tile_size.2 as f32 * lod0.voxel_size.2;
1011            self.cached_ideal_lod = self.select_lod_for_tile(vol_center, typical, camera_info);
1012        }
1013
1014        // Per-tile LOD selection: refine a tile only when its ideal LOD is finer than its
1015        // current LOD. Tiles close to the camera get refined; distant tiles stay coarse.
1016        while let Some((lod, tz, ty, tx)) = stack.pop() {
1017            let aabb = self.compute_tile_aabb(lod, (tz, ty, tx));
1018
1019            // Frustum cull — skip tiles the camera cannot see.
1020            let margin = (aabb.max - aabb.min) * 0.05;
1021            if !camera_info.frustum.contains_aabb_xy(aabb.min - margin, aabb.max + margin) {
1022                continue;
1023            }
1024
1025            if lod == 0 {
1026                // Already at finest LOD — just mark it.
1027                self.visible_tile_keys.insert(TileKey::new(lod, 0, tz, ty, tx));
1028                continue;
1029            }
1030
1031            // How fine does the camera need this tile? Use the tile's own centre + X-extent.
1032            let tile_center = (aabb.min + aabb.max) * 0.5;
1033            let tile_size   = aabb.max.x - aabb.min.x;
1034            let ideal_for_tile = self.select_lod_for_tile(tile_center, tile_size, camera_info);
1035
1036            if lod <= ideal_for_tile {
1037                // Current LOD is fine enough (camera is far away enough) — use this tile.
1038                self.visible_tile_keys.insert(TileKey::new(lod, 0, tz, ty, tx));
1039            } else {
1040                // Need finer data — push children of the next finer LOD.
1041                let child_lod    = lod - 1;
1042                let parent_grid  = self.levels[lod].grid_size();
1043                let child_grid   = self.levels[child_lod].grid_size();
1044                let cpp_z = child_grid.0.div_ceil(parent_grid.0).max(1);
1045                let cpp_y = child_grid.1.div_ceil(parent_grid.1).max(1);
1046                let cpp_x = child_grid.2.div_ceil(parent_grid.2).max(1);
1047                for dz in 0..cpp_z {
1048                    for dy in 0..cpp_y {
1049                        for dx in 0..cpp_x {
1050                            let (cz, cy, cx) = (tz * cpp_z + dz, ty * cpp_y + dy, tx * cpp_x + dx);
1051                            if cz < child_grid.0 && cy < child_grid.1 && cx < child_grid.2 {
1052                                stack.push((child_lod, cz, cy, cx));
1053                            }
1054                        }
1055                    }
1056                }
1057            }
1058        }
1059    }

1061    pub(crate) fn prepare_volume(
1062        &mut self,
1063        queue: &Queue,
1064        frame_number: u64,
1065        camera_info: &crate::visual::CameraInfo,
1066    ) {
1067        self.stats = PrepareStats { frame: frame_number, ..Default::default() };
1068        self.frame_counter = frame_number;
1069        self.update_visible_tiles_volume(camera_info);
1070        // Touch LRU for displayed slots (at desired_t — anything else
1071        // is shader-filtered out by the page-table t-compare).
1072        let target = self.desired_t;
1073        for spatial in &self.visible_tile_keys {
1074            let key = TileKey { lod_level: spatial.lod_level, t: target,
1075                                z: spatial.z, y: spatial.y, x: spatial.x };
1076            if self.slot_map.contains_key(&key) {
1077                self.lru_map.insert(key, frame_number);
1078            }
1079        }
1080        self.process_pending_chunks(queue);
1081        self.refresh_page_table(queue);
1082        // Publish what we want; the loader (whoever wired one up) picks
1083        // it up on its next poll.
1084        self.publish_wanted();
1085        self.stats.slot_map_len = self.slot_map.len();
1086        self.stats.pending_len = self.pending_chunks.lock().unwrap().len();
1087    }

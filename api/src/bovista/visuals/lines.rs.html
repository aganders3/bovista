6/// A vertex for line rendering with position and color
7#[repr(C)]
8#[derive(Copy, Clone, Debug, Pod, Zeroable)]
9pub struct LineVertex {
10    pub position: [f32; 3],
11    pub color: [f32; 3],
12}
13
14impl LineVertex {
15    const ATTRIBS: [wgpu::VertexAttribute; 2] =
16        wgpu::vertex_attr_array![0 => Float32x3, 1 => Float32x3];
17
18    pub fn desc() -> wgpu::VertexBufferLayout<'static> {
19        wgpu::VertexBufferLayout {
20            array_stride: std::mem::size_of::<LineVertex>() as wgpu::BufferAddress,
21            step_mode: wgpu::VertexStepMode::Vertex,
22            attributes: &Self::ATTRIBS,
23        }
24    }
25}

37impl Lines {
38    /// Create a new Lines from vertex pairs
39    /// Each pair of vertices defines a line segment
40    pub fn new(
41        device: &wgpu::Device,
42        surface_format: wgpu::TextureFormat,
43        camera_bind_group_layout: &wgpu::BindGroupLayout,
44        vertices: Vec<LineVertex>,
45    ) -> Self {
46        Self::with_shader(
47            device,
48            surface_format,
49            camera_bind_group_layout,
50            vertices,
51            include_str!("../shaders/lines.wgsl"),
52        )
53    }

55    /// Create a new Lines with a custom shader
56    pub fn with_shader(
57        device: &wgpu::Device,
58        surface_format: wgpu::TextureFormat,
59        camera_bind_group_layout: &wgpu::BindGroupLayout,
60        vertices: Vec<LineVertex>,
61        shader_source: &str,
62    ) -> Self {
63        let vertex_count = vertices.len() as u32;
64
65        let vertex_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
66            label: Some("Lines Vertex Buffer"),
67            contents: bytemuck::cast_slice(&vertices),
68            usage: wgpu::BufferUsages::VERTEX,
69        });
70
71        let shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
72            label: Some("Lines Shader"),
73            source: wgpu::ShaderSource::Wgsl(shader_source.into()),
74        });
75
76        let pipeline_layout = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
77            label: Some("Lines Pipeline Layout"),
78            bind_group_layouts: &[camera_bind_group_layout],
79            push_constant_ranges: &[],
80        });
81
82        let render_pipeline = device.create_render_pipeline(&wgpu::RenderPipelineDescriptor {
83            label: Some("Lines Render Pipeline"),
84            layout: Some(&pipeline_layout),
85            vertex: wgpu::VertexState {
86                module: &shader,
87                entry_point: Some("vs_main"),
88                buffers: &[LineVertex::desc()],
89                compilation_options: Default::default(),
90            },
91            fragment: Some(wgpu::FragmentState {
92                module: &shader,
93                entry_point: Some("fs_main"),
94                targets: &[Some(wgpu::ColorTargetState {
95                    format: surface_format,
96                    blend: Some(wgpu::BlendState::REPLACE),
97                    write_mask: wgpu::ColorWrites::ALL,
98                })],
99                compilation_options: Default::default(),
100            }),
101            primitive: wgpu::PrimitiveState {
102                topology: wgpu::PrimitiveTopology::LineList,
103                strip_index_format: None,
104                front_face: wgpu::FrontFace::Ccw,
105                cull_mode: None,
106                unclipped_depth: false,
107                polygon_mode: wgpu::PolygonMode::Fill,
108                conservative: false,
109            },
110            depth_stencil: Some(wgpu::DepthStencilState {
111                format: wgpu::TextureFormat::Depth24PlusStencil8,
112                depth_write_enabled: true,
113                depth_compare: wgpu::CompareFunction::Less,
114                stencil: wgpu::StencilState::default(),
115                bias: wgpu::DepthBiasState::default(),
116            }),
117            multisample: wgpu::MultisampleState {
118                count: 1,
119                mask: !0,
120                alpha_to_coverage_enabled: false,
121            },
122            multiview: None,
123            cache: None,
124        });
125
126        Self {
127            vertex_buffer,
128            vertex_count,
129            render_pipeline,
130            transform: Transform::identity(),
131            visible: true,
132            name: "Lines".to_string(),
133        }
134    }

136    /// Create a test wireframe cube
137    pub fn test_cube(
138        device: &wgpu::Device,
139        surface_format: wgpu::TextureFormat,
140        camera_bind_group_layout: &wgpu::BindGroupLayout,
141    ) -> Self {
142        let color = [1.0, 1.0, 1.0]; // White lines
143
144        // Define the 8 corners of a cube
145        let corners = [
146            [-1.0, -1.0, -1.0], // 0
147            [1.0, -1.0, -1.0],  // 1
148            [1.0, 1.0, -1.0],   // 2
149            [-1.0, 1.0, -1.0],  // 3
150            [-1.0, -1.0, 1.0],  // 4
151            [1.0, -1.0, 1.0],   // 5
152            [1.0, 1.0, 1.0],    // 6
153            [-1.0, 1.0, 1.0],   // 7
154        ];
155
156        // Define the 12 edges of a cube (as pairs of corner indices)
157        let edges = [
158            (0, 1), (1, 2), (2, 3), (3, 0), // Bottom face
159            (4, 5), (5, 6), (6, 7), (7, 4), // Top face
160            (0, 4), (1, 5), (2, 6), (3, 7), // Vertical edges
161        ];
162
163        let mut vertices = Vec::new();
164        for (start, end) in edges.iter() {
165            vertices.push(LineVertex {
166                position: corners[*start],
167                color,
168            });
169            vertices.push(LineVertex {
170                position: corners[*end],
171                color,
172            });
173        }
174
175        Self::new(device, surface_format, camera_bind_group_layout, vertices)
176    }

178    /// Create a 3D axis helper (XYZ axes)

180        device: &wgpu::Device,
181        surface_format: wgpu::TextureFormat,

191            LineVertex {
192                position: [length, 0.0, 0.0],
193                color: [1.0, 0.0, 0.0],
194            },
195            // Y axis (green)

197                position: [0.0, 0.0, 0.0],

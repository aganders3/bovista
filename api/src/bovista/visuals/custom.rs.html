5/// A custom visual that allows users to provide their own WGSL shader
6///
7/// # Example
8///
9/// ```ignore
10/// let shader = r#"
11///     struct Uniforms {
12///         view_proj: mat4x4<f32>,
13///     }
14///
15///     @group(0) @binding(0)
16///     var<uniform> uniforms: Uniforms;
17///
18///     struct VertexInput {
19///         @location(0) position: vec3<f32>,
20///         @location(1) color: vec4<f32>,
21///     }
22///
23///     struct VertexOutput {
24///         @builtin(position) clip_position: vec4<f32>,
25///         @location(0) color: vec4<f32>,
26///     }
27///
28///     @vertex
29///     fn vs_main(input: VertexInput) -> VertexOutput {
30///         var output: VertexOutput;
31///         output.clip_position = uniforms.view_proj * vec4<f32>(input.position, 1.0);
32///         output.color = input.color;
33///         return output;
34///     }
35///
36///     @fragment
37///     fn fs_main(input: VertexOutput) -> @location(0) vec4<f32> {
38///         return input.color;
39///     }
40/// "#;
41///
42/// let visual = Custom::new(
43///     device,
44///     surface_format,
45///     camera_bind_group_layout,
46///     shader,
47///     vertex_data,
48///     vertex_layout,
49///     wgpu::PrimitiveTopology::TriangleList,
50/// );
51/// ```
52///
53/// # Shader Requirements
54///
55/// Your shader MUST:
56/// - Have a camera uniform at `@group(0) @binding(0)` with structure:
57///   ```wgsl
58///   struct Uniforms {
59///       view_proj: mat4x4<f32>,
60///   }
61///   ```
62/// - Have a vertex entry point named `vs_main`
63/// - Have a fragment entry point named `fs_main`
64/// - Match the vertex layout you provide
65///
66/// # Cross-Platform Support
67///
68/// Custom shaders work on **both native and WebAssembly/browser** targets!
69/// WGSL compilation happens at runtime via the WebGPU API, so your custom
70/// shaders will work in the browser just as well as native applications.
71pub struct Custom {
72    vertex_buffer: wgpu::Buffer,
73    vertex_count: u32,
74    render_pipeline: wgpu::RenderPipeline,
75    transform: Transform,
76    visible: bool,
77    name: String,
78}

80impl Custom {
81    /// Create a new custom visual with a user-provided WGSL shader
82    ///
83    /// # Arguments
84    ///
85    /// * `device` - The GPU device
86    /// * `surface_format` - The surface format for rendering
87    /// * `camera_bind_group_layout` - The camera bind group layout (from Renderer)
88    /// * `shader_source` - WGSL shader source code as a string
89    /// * `vertex_data` - Raw vertex data as bytes
90    /// * `vertex_layout` - Description of the vertex buffer layout
91    /// * `topology` - Primitive topology (PointList, LineList, TriangleList, etc.)
92    pub fn new(
93        device: &wgpu::Device,
94        surface_format: wgpu::TextureFormat,
95        camera_bind_group_layout: &wgpu::BindGroupLayout,
96        shader_source: &str,
97        vertex_data: &[u8],
98        vertex_layout: VertexBufferLayout,
99        topology: wgpu::PrimitiveTopology,
100    ) -> Result<Self, String> {
101        let vertex_count = (vertex_data.len() as u64 / vertex_layout.stride) as u32;
102
103        let vertex_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
104            label: Some("Custom Visual Vertex Buffer"),
105            contents: vertex_data,
106            usage: wgpu::BufferUsages::VERTEX,
107        });
108
109        // Compile the user-provided shader
110        let shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
111            label: Some("Custom Visual Shader"),
112            source: wgpu::ShaderSource::Wgsl(shader_source.into()),
113        });
114
115        let pipeline_layout = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
116            label: Some("Custom Visual Pipeline Layout"),
117            bind_group_layouts: &[camera_bind_group_layout],
118            push_constant_ranges: &[],
119        });
120
121        let wgpu_vertex_layout = vertex_layout.to_wgpu();
122
123        let render_pipeline = device.create_render_pipeline(&wgpu::RenderPipelineDescriptor {
124            label: Some("Custom Visual Render Pipeline"),
125            layout: Some(&pipeline_layout),
126            vertex: wgpu::VertexState {
127                module: &shader,
128                entry_point: Some("vs_main"),
129                buffers: &[wgpu_vertex_layout],
130                compilation_options: Default::default(),
131            },
132            fragment: Some(wgpu::FragmentState {
133                module: &shader,
134                entry_point: Some("fs_main"),
135                targets: &[Some(wgpu::ColorTargetState {
136                    format: surface_format,
137                    blend: Some(wgpu::BlendState::ALPHA_BLENDING),
138                    write_mask: wgpu::ColorWrites::ALL,
139                })],
140                compilation_options: Default::default(),
141            }),
142            primitive: wgpu::PrimitiveState {
143                topology,
144                strip_index_format: None,
145                front_face: wgpu::FrontFace::Ccw,
146                cull_mode: None,
147                unclipped_depth: false,
148                polygon_mode: wgpu::PolygonMode::Fill,
149                conservative: false,
150            },
151            depth_stencil: Some(wgpu::DepthStencilState {
152                format: wgpu::TextureFormat::Depth24PlusStencil8,
153                depth_write_enabled: true,
154                depth_compare: wgpu::CompareFunction::Less,
155                stencil: wgpu::StencilState::default(),
156                bias: wgpu::DepthBiasState::default(),
157            }),
158            multisample: wgpu::MultisampleState {
159                count: 1,
160                mask: !0,
161                alpha_to_coverage_enabled: false,
162            },
163            multiview: None,
164            cache: None,
165        });
166
167        Ok(Self {
168            vertex_buffer,
169            vertex_count,
170            render_pipeline,
171            transform: Transform::identity(),
172            visible: true,
173            name: "Custom Visual".to_string(),
174        })
175    }
176
177    pub fn set_name(&mut self, name: String) {
178        self.name = name;
179    }
180}

182impl Visual for Custom {
183    fn prepare(&mut self, _device: &wgpu::Device, _queue: &wgpu::Queue, _camera_info: &crate::visual::CameraInfo) {
184        // Custom visuals don't need any per-frame preparation currently
185        // In the future, this could handle dynamic uniform updates
186    }
187
188    fn render(&self, render_pass: &mut RenderPass) {
189        render_pass.set_pipeline(&self.render_pipeline);
190        render_pass.set_vertex_buffer(0, self.vertex_buffer.slice(..));
191        render_pass.draw(0..self.vertex_count, 0..1);
192    }
193
194    fn set_transform(&mut self, transform: Transform) {
195        self.transform = transform;
196    }
197
198    fn transform(&self) -> &Transform {
199        &self.transform
200    }
201
202    fn is_visible(&self) -> bool {
203        self.visible
204    }
205
206    fn set_visible(&mut self, visible: bool) {
207        self.visible = visible;
208    }
209
210    fn name(&self) -> &str {
211        &self.name
212    }

1// Virtual-texture page table backed by a wgpu R32Uint 2D-array texture.
2//
3// Layout:
4//   - One array layer per LOD level.
5//   - Width  = min(max_linear, max_texture_dimension_2d).
6//   - Height = ceil(max_linear / width).
7//   - Each texel encodes a single tile:
8//       bit 31     : resident
9//       bits 29-30 : atlas_id (2 bits, 0..=3) — which of up to MAX_ATLAS_COUNT
10//                    physical atlas textures holds this tile's voxel data
11//       bits 16-28 : timepoint `t` (13 bits, 0..=8191)
12//       bits  0-15 : atlas slot index (16 bits, 0..=65535)
13//     The shader compares the texel's t against a `desired_t` uniform; a
14//     mismatch is treated identically to non-resident, so the LOD fallback
15//     walks to a coarser tile rather than displaying stale-t data.
16//
17// Coordinate mapping (must match the shader):
18//   linear_index = tz * gy * gx + ty * gx + tx
19//   texel_x = linear_index % width
20//   texel_y = linear_index / width
21//
22// The page table is sampled via `textureLoad` (exact-index, no filtering)
23// while the atlas is read via `textureSampleLevel` (Nearest filter for
24// data fidelity, but with linear blending available if we ever want it).
25
26use crate::visuals::virtual_texture::LodLevelConfig;
27
28pub struct PageTable {
29    pub texture: wgpu::Texture,
30    pub texture_view: wgpu::TextureView,
31    /// Width of the 2D page table texture (used for linear→2D coord mapping).
32    pub width: u32,
33    // (gx, gy, gz) for each LOD
34    lod_grids: Vec<(u32, u32, u32)>,
35}
36
37impl PageTable {
38    pub fn new(device: &wgpu::Device, levels: &[LodLevelConfig]) -> Self {
39        assert!(!levels.is_empty(), "PageTable requires at least one LOD level");
40
41        let mut lod_grids = Vec::with_capacity(levels.len());
42        let mut max_linear = 0u32;
43
44        for level in levels {
45            let (gz, gy, gx) = level.grid_size();
46            lod_grids.push((gx, gy, gz));
47            max_linear = max_linear.max(gz * gy * gx);
48        }
49
50        let max_dim = device.limits().max_texture_dimension_2d;
51        let pt_width  = max_linear.min(max_dim);
52        let pt_height = max_linear.div_ceil(pt_width);
53        let num_lods  = levels.len() as u32;
54
55        let texture = device.create_texture(&wgpu::TextureDescriptor {
56            label: Some("VT Page Table"),
57            size: wgpu::Extent3d {
58                width: pt_width,
59                height: pt_height,
60                // Pad to >= 2 layers so wgpu-hal-GL creates a real
61                // GL_TEXTURE_2D_ARRAY object — its heuristic maps a
62                // single-layer descriptor to plain GL_TEXTURE_2D, which
63                // can't be sampled through a D2Array binding. The
64                // shader's lod_count uniform bounds reads to the real
65                // LOD count, so the padding layer is never read.
66                depth_or_array_layers: num_lods.max(2),
67            },
68            mip_level_count: 1,
69            sample_count: 1,
70            dimension: wgpu::TextureDimension::D2,
71            format: wgpu::TextureFormat::R32Uint,
72            usage: wgpu::TextureUsages::TEXTURE_BINDING | wgpu::TextureUsages::COPY_DST,
73            view_formats: &[],
74        });
75
76        let texture_view = texture.create_view(&wgpu::TextureViewDescriptor {
77            dimension: Some(wgpu::TextureViewDimension::D2Array),
78            ..Default::default()
79        });
80
81        Self { texture, texture_view, width: pt_width, lod_grids }
82    }
83
84    /// Mark a tile as resident at the given atlas slot, tagged with the
85    /// timepoint `t` whose data the slot currently holds. The shader
86    /// compares `t` to its `desired_t` uniform; mismatched-t reads are
87    /// treated as non-resident, so stale-t entries left behind after a
88    /// scrub gracefully fall through to coarser LODs without needing
89    /// explicit clearing on every desired_t change.
90    #[allow(clippy::too_many_arguments)]
91    pub fn update(&self, queue: &wgpu::Queue, lod: usize, tz: u32, ty: u32, tx: u32,
92                  atlas_id: u32, t: u32, slot: u32) {
93        let (gx, gy, _gz) = self.lod_grids[lod];
94        let linear = tz * gy * gx + ty * gx + tx;
95        let value: u32 = (1u32 << 31)
96                       | ((atlas_id & 0x3) << 29)
97                       | ((t & 0x1FFF) << 16)
98                       | (slot & 0xFFFF);
99        self.write_texel(queue, lod as u32, linear, value);
100    }
101
102    /// Mark a tile as non-resident (clear its entry).
103    pub fn clear(&self, queue: &wgpu::Queue, lod: usize, tz: u32, ty: u32, tx: u32) {
104        let (gx, gy, _gz) = self.lod_grids[lod];
105        let linear = tz * gy * gx + ty * gx + tx;
106        self.write_texel(queue, lod as u32, linear, 0u32);
107    }
108
109    fn write_texel(&self, queue: &wgpu::Queue, layer: u32, linear: u32, value: u32) {
110        let px = linear % self.width;
111        let py = linear / self.width;
112        queue.write_texture(
113            wgpu::TexelCopyTextureInfo {
114                texture: &self.texture,
115                mip_level: 0,
116                origin: wgpu::Origin3d { x: px, y: py, z: layer },
117                aspect: wgpu::TextureAspect::All,
118            },
119            &value.to_ne_bytes(),
120            wgpu::TexelCopyBufferLayout {
121                offset: 0,
122                bytes_per_row: Some(self.width * 4),
123                rows_per_image: None,
124            },
125            wgpu::Extent3d { width: 1, height: 1, depth_or_array_layers: 1 },
126        );
127    }
128}

1//! Point cloud rendering for 3D scatter plots and particle systems.
2//!
3//! This module provides [`Points`] for rendering collections of colored points in 3D space.
4//! Each point has a position and RGB color, making it ideal for:
5//! - Scientific data visualization (scatter plots)
6//! - Particle systems
7//! - Point cloud data from sensors (LiDAR, photogrammetry)
8//!
9//! # Examples
10//!
11//! ## Creating a Point Cloud from Vertices
12//!
13//! ```rust,no_run
14//! use bovista::{Points, Renderer};
15//! use bovista::visuals::points::PointVertex;
16//! # async fn example(renderer: Renderer) {
17//!
18//! // Create colored vertices
19//! let vertices = vec![
20//!     PointVertex { position: [0.0, 0.0, 0.0], color: [1.0, 0.0, 0.0] }, // Red
21//!     PointVertex { position: [1.0, 0.0, 0.0], color: [0.0, 1.0, 0.0] }, // Green
22//!     PointVertex { position: [0.0, 1.0, 0.0], color: [0.0, 0.0, 1.0] }, // Blue
23//! ];
24//!
25//! let points = Points::new(
26//!     renderer.device(),
27//!     renderer.surface_format(),
28//!     renderer.camera_bind_group_layout(),
29//!     vertices
30//! );
31//! # }
32//! ```
33//!
34//! ## Live Demo
35//!
36//! **[🌐 Try it in your browser →](https://yourdomain.com/bovista/examples/point_cloud.html)**
37//!
38//! Runs directly in any browser with WebGPU support (Chrome 113+, Safari 18+).
39//! No installation required!
40//!
41//! ## Python Usage
42//!
43//! ```python
44//! import bovista as bv
45//! import numpy as np
46//!
47//! viewer = bv.PyViewer(800, 600)
48//! viewer.initialize()
49//!
50//! # Generate random point cloud
51//! n_points = 10000

53//! colors = np.random.rand(n_points, 1, 3).astype(np.float32)
54//!
55//! points = bv.PyPoints.from_numpy(viewer, positions, colors)

87/// A vertex in a point cloud with position and RGB color.
88///
89/// Each vertex is 24 bytes (2×[f32; 3]) and can be directly uploaded to GPU memory.
90/// Marked with [`Pod`] and [`Zeroable`] for safe zero-copy transfer.
91///
92/// # Example
93///
94/// ```rust
95/// use bovista::visuals::points::PointVertex;
96///
97/// let vertex = PointVertex {
98///     position: [1.0, 2.0, 3.0],  // World-space position
99///     color: [1.0, 0.5, 0.0],     // RGB in 0-1 range (orange)
100/// };
101/// ```
102#[repr(C)]
103#[derive(Copy, Clone, Debug, Pod, Zeroable)]
104pub struct PointVertex {
105    /// World-space position (X, Y, Z)
106    pub position: [f32; 3],
107    /// RGB color (each component 0.0 to 1.0)
108    pub color: [f32; 3],
109}
110
111impl PointVertex {
112    const ATTRIBS: [wgpu::VertexAttribute; 2] =
113        wgpu::vertex_attr_array![0 => Float32x3, 1 => Float32x3];
114
115    pub fn desc() -> wgpu::VertexBufferLayout<'static> {
116        wgpu::VertexBufferLayout {
117            array_stride: std::mem::size_of::<PointVertex>() as wgpu::BufferAddress,
118            step_mode: wgpu::VertexStepMode::Vertex,
119            attributes: &Self::ATTRIBS,
120        }
121    }
122}

124/// Visual for rendering point clouds in 3D space.
125///
126/// `Points` renders a collection of colored points using GPU acceleration.
127/// Each point is rendered as a screen-space circle with configurable size.
128///
129/// # GPU Resources
130///
131/// - **Vertex Buffer**: Stores all point positions and colors (immutable after creation)
132/// - **Render Pipeline**: Point rendering shader with depth testing
133/// - **Memory**: ~24 bytes per point (position + color)
134///
135/// # Performance
136///
137/// Can efficiently render millions of points per frame on modern GPUs.
138/// Rendering cost is O(n) where n is the number of visible points.
139///
140/// # Examples
141///
142/// ## Basic Usage
143///
144/// ```rust,no_run
145/// # use bovista::{Points, Renderer, Scene};
146/// # use bovista::visuals::points::PointVertex;
147/// # use std::sync::{Arc, Mutex};
148/// # async fn example(renderer: Renderer) {
149/// let vertices = vec![
150///     PointVertex { position: [0.0, 0.0, 0.0], color: [1.0, 0.0, 0.0] },
151///     PointVertex { position: [1.0, 1.0, 1.0], color: [0.0, 1.0, 0.0] },
152/// ];
153///
154/// let points = Points::new(
155///     renderer.device(),
156///     renderer.surface_format(),
157///     renderer.camera_bind_group_layout(),
158///     vertices
159/// );
160///
161/// let mut scene = Scene::new();
162/// scene.add(Arc::new(Mutex::new(points)));
163/// # }
164/// ```
165///
166/// ## Try it yourself
167///
168/// Run the complete interactive example:
169///
170/// ```bash
171/// cargo run --example point_cloud
172/// ```
173pub struct Points {
174    vertex_buffer: wgpu::Buffer,
175    vertex_count: u32,
176    render_pipeline: wgpu::RenderPipeline,
177    transform: Transform,
178    visible: bool,
179    name: String,
180}

182impl Points {
183    /// Creates a new point cloud visual from vertex data.
184    ///
185    /// # Arguments
186    ///
187    /// * `device` - GPU device for creating buffers and pipelines
188    /// * `surface_format` - Target surface format for rendering
189    /// * `camera_bind_group_layout` - Shared camera uniform layout
190    /// * `vertices` - Vector of point vertices (position + color)
191    ///
192    /// # Note
193    ///
194    /// Vertex data is uploaded to GPU immediately and cannot be modified after creation.
195    /// To update points, create a new `Points` instance.
196    ///
197    /// # Example
198    ///
199    /// ```rust,no_run
200    /// # use bovista::{Points, Renderer};
201    /// # use bovista::visuals::points::PointVertex;
202    /// # async fn example(renderer: Renderer) {
203    /// let vertices = vec![
204    ///     PointVertex { position: [0.0, 0.0, 0.0], color: [1.0, 1.0, 1.0] },
205    ///     // ... more vertices
206    /// ];
207    ///
208    /// let points = Points::new(
209    ///     renderer.device(),
210    ///     renderer.surface_format(),
211    ///     renderer.camera_bind_group_layout(),
212    ///     vertices
213    /// );
214    /// # }
215    /// ```
216    pub fn new(
217        device: &wgpu::Device,
218        surface_format: wgpu::TextureFormat,
219        camera_bind_group_layout: &wgpu::BindGroupLayout,
220        vertices: Vec<PointVertex>,
221    ) -> Self {
222        Self::with_shader(
223            device,
224            surface_format,
225            camera_bind_group_layout,
226            vertices,
227            include_str!("../shaders/point_cloud.wgsl"),
228        )
229    }
230
231    /// Create a new Points with a custom shader
232    pub fn with_shader(
233        device: &wgpu::Device,
234        surface_format: wgpu::TextureFormat,
235        camera_bind_group_layout: &wgpu::BindGroupLayout,
236        vertices: Vec<PointVertex>,
237        shader_source: &str,
238    ) -> Self {
239        let vertex_count = vertices.len() as u32;
240
241        let vertex_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
242            label: Some("Points Vertex Buffer"),
243            contents: bytemuck::cast_slice(&vertices),
244            usage: wgpu::BufferUsages::VERTEX,
245        });
246
247        let shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
248            label: Some("Points Shader"),
249            source: wgpu::ShaderSource::Wgsl(shader_source.into()),
250        });
251
252        let pipeline_layout = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
253            label: Some("Points Pipeline Layout"),
254            bind_group_layouts: &[camera_bind_group_layout],
255            push_constant_ranges: &[],
256        });
257
258        let render_pipeline = device.create_render_pipeline(&wgpu::RenderPipelineDescriptor {
259            label: Some("Points Render Pipeline"),
260            layout: Some(&pipeline_layout),
261            vertex: wgpu::VertexState {
262                module: &shader,
263                entry_point: Some("vs_main"),
264                buffers: &[PointVertex::desc()],
265                compilation_options: Default::default(),
266            },
267            fragment: Some(wgpu::FragmentState {
268                module: &shader,
269                entry_point: Some("fs_main"),
270                targets: &[Some(wgpu::ColorTargetState {
271                    format: surface_format,
272                    blend: Some(wgpu::BlendState::REPLACE),
273                    write_mask: wgpu::ColorWrites::ALL,
274                })],
275                compilation_options: Default::default(),
276            }),
277            primitive: wgpu::PrimitiveState {
278                topology: wgpu::PrimitiveTopology::PointList,
279                strip_index_format: None,
280                front_face: wgpu::FrontFace::Ccw,
281                cull_mode: None,
282                unclipped_depth: false,
283                polygon_mode: wgpu::PolygonMode::Fill,
284                conservative: false,
285            },
286            depth_stencil: Some(wgpu::DepthStencilState {
287                format: wgpu::TextureFormat::Depth24PlusStencil8,
288                depth_write_enabled: true,
289                depth_compare: wgpu::CompareFunction::Less,
290                stencil: wgpu::StencilState::default(),
291                bias: wgpu::DepthBiasState::default(),
292            }),
293            multisample: wgpu::MultisampleState {
294                count: 1,
295                mask: !0,
296                alpha_to_coverage_enabled: false,
297            },
298            multiview: None,
299            cache: None,
300        });
301
302        Self {
303            vertex_buffer,
304            vertex_count,
305            render_pipeline,
306            transform: Transform::identity(),
307            visible: true,
308            name: "Points".to_string(),
309        }
310    }

312    /// Creates a test cube of colored points for demos and testing.
313    ///
314    /// Generates a `size × size × size` cube of points centered at the origin,
315    /// spanning from -1 to +1 in each dimension. Points are colored based on
316    /// their position (red = X, green = Y, blue = Z), creating a rainbow gradient.
317    ///
318    /// # Arguments
319    ///
320    /// * `size` - Number of points along each axis (total points = size³)
321    ///
322    /// # Examples
323    ///
324    /// ```rust,no_run
325    /// # use bovista::{Points, Renderer};
326    /// # async fn example(renderer: Renderer) {
327    /// // Create 20×20×20 = 8,000 colored points
328    /// let cube = Points::test_cube(
329    ///     renderer.device(),
330    ///     renderer.surface_format(),
331    ///     renderer.camera_bind_group_layout(),
332    ///     20
333    /// );
334    /// # }
335    /// ```
336    ///
337    /// # Performance
338    ///
339    /// - `size = 10`: 1,000 points (~24 KB)
340    /// - `size = 50`: 125,000 points (~3 MB)
341    /// - `size = 100`: 1,000,000 points (~24 MB)
342    ///
343    /// # Try it Live
344    ///
345    /// **[🌐 Interactive Demo](https://yourdomain.com/bovista/examples/point_cloud.html)**
346    pub fn test_cube(
347        device: &wgpu::Device,
348        surface_format: wgpu::TextureFormat,
349        camera_bind_group_layout: &wgpu::BindGroupLayout,
350        size: u32,
351    ) -> Self {
352        let mut vertices = Vec::new();
353        let step = 2.0 / size as f32;
354
355        for i in 0..size {
356            for j in 0..size {
357                for k in 0..size {
358                    let x = -1.0 + i as f32 * step;
359                    let y = -1.0 + j as f32 * step;
360                    let z = -1.0 + k as f32 * step;
361
362                    // Color based on position (normalized to 0-1)
363                    let color = [(x + 1.0) / 2.0, (y + 1.0) / 2.0, (z + 1.0) / 2.0];
364
365                    vertices.push(PointVertex {
366                        position: [x, y, z],
367                        color,
368                    });
369                }
370            }
371        }
372
373        Self::new(device, surface_format, camera_bind_group_layout, vertices)
374    }

381impl Visual for Points {
382    fn prepare(&mut self, _device: &wgpu::Device, _queue: &wgpu::Queue, _camera_info: &crate::visual::CameraInfo) {
383        // Points don't need any per-frame preparation currently
384        // In the future, this could handle dynamic updates to vertex data
385    }
386
387    fn render(&self, render_pass: &mut RenderPass) {
388        render_pass.set_pipeline(&self.render_pipeline);
389        render_pass.set_vertex_buffer(0, self.vertex_buffer.slice(..));
390        render_pass.draw(0..self.vertex_count, 0..1);
391    }
392
393    fn set_transform(&mut self, transform: Transform) {
394        self.transform = transform;
395    }

399    }

406        self.visible = visible;
407    }

410        &self.name

11/// Additive blend (`src + dst` on both color and alpha). Correct only on
12/// premultiplied-alpha output (which both the image and volume shaders now
13/// produce). Used for the `BlendMode::Additive` pipeline variant, which is the
14/// basis for order-independent multi-channel compositing.
15pub const ADDITIVE_BLENDING: wgpu::BlendState = wgpu::BlendState {
16    color: wgpu::BlendComponent {
17        src_factor: wgpu::BlendFactor::One,
18        dst_factor: wgpu::BlendFactor::One,
19        operation: wgpu::BlendOperation::Add,
20    },
21    alpha: wgpu::BlendComponent {
22        src_factor: wgpu::BlendFactor::One,
23        dst_factor: wgpu::BlendFactor::One,
24        operation: wgpu::BlendOperation::Add,
25    },
26};

28/// Response containing tile/chunk data.
29///
30/// Shape uses the numpy convention: `z_shape`, `y_shape`, `x_shape` are voxel
31/// counts along each axis. The struct field order is z/y/x to match how the
32/// data is typically indexed (slowest-varying first). For a 2D image the
33/// `z_shape` is 1.
34#[derive(Debug, Clone)]
35pub struct TileData {
36    /// Raw voxel data as bytes (layout determined by `format`)
37    pub data: Vec<u8>,
38
39    /// Voxel count along Z (slowest-varying axis).
40    pub z_shape: u32,
41    /// Voxel count along Y.
42    pub y_shape: u32,
43    /// Voxel count along X (fastest-varying axis).
44    pub x_shape: u32,
45
46    /// GPU texture format for this data (determines bytes-per-voxel)
47    pub format: wgpu::TextureFormat,
48}
49
50impl TileData {
51    /// Bytes per voxel for the stored format
52    pub fn bytes_per_voxel(&self) -> u32 {
53        match self.format {
54            wgpu::TextureFormat::R8Unorm => 1,
55            wgpu::TextureFormat::R16Float => 2,
56            wgpu::TextureFormat::R32Float => 4,
57            _ => 1,
58        }
59    }
60}

62/// Axis-aligned bounding box in 3D space
63#[derive(Debug, Clone, Copy)]
64pub struct AABB {
65    pub min: Vec3,
66    pub max: Vec3,
67}
68
69impl AABB {
70    pub fn new(min: Vec3, max: Vec3) -> Self {
71        Self { min, max }
72    }
73
74    pub fn center(&self) -> Vec3 {
75        (self.min + self.max) * 0.5
76    }
77
78    pub fn size(&self) -> Vec3 {
79        self.max - self.min
80    }

83/// A single tile of image data with associated GPU resources
84///
85/// Tiles are the fundamental rendering unit. A simple image has one tile,
86/// while a large tiled image has many tiles. Each tile has its own texture
87/// and geometry, but shares the rendering pipeline with other tiles.
88pub struct Tile {
89    /// Unique identifier for this tile
90    pub key: TileKey,
91
92    /// GPU texture containing the voxel data
93    pub texture: Arc<Texture>,
94
95    /// Texture view for binding
96    pub texture_view: Arc<TextureView>,
97
98    /// World-space bounding box for this tile
99    pub bounds: AABB,
100
101    /// Size of this tile in voxels (Z, Y, X)
102    pub voxel_count: (u32, u32, u32),
103
104    /// Physical size of each voxel in world units (Z, Y, X)
105    pub voxel_size: (f32, f32, f32),
106
107    /// LOD level (0 = highest resolution, higher = coarser)
108    pub lod_level: usize,
109
110    /// Last frame this tile was accessed (for LRU eviction)
111    pub last_used_frame: u64,
112
113    /// Bind group for this tile's resources (texture, sampler, uniforms)
114    /// Created on-demand during rendering
115    pub bind_group: Option<Arc<BindGroup>>,
116
117    /// Vertex buffer containing the tile's geometry (slice intersection polygon)
118    /// Updated when slice plane changes
119    pub vertex_buffer: Option<Buffer>,
120
121    /// Number of vertices currently in the vertex buffer
122    pub vertex_count: u32,
123
124    /// Index buffer for rendering the tile geometry
125    pub index_buffer: Option<Buffer>,
126
127    /// Whether this tile is a placeholder (no real data loaded yet)
128    pub is_placeholder: bool,
129
130    /// Debug visualization color (for debugging LOD levels or spatial structure)
131    pub debug_color: Option<[f32; 3]>,
132}

225/// Per-LOD data in the VT uniforms block.
226/// Layout must match `VTLodInfo` in `virtual_tile.wgsl`.
227#[repr(C)]
228#[derive(Copy, Clone, Debug, Pod, Zeroable)]
229pub struct VTLodInfo {
230    /// Number of tiles in (x, y, z) at this LOD (ceil).
231    pub grid_dims: [u32; 3],
232    pub _pad: u32,
233    /// tile_size / volume_size per axis (x, y, z).
234    /// Used for exact sub-tile UV computation independent of grid rounding.
235    pub tile_scale: [f32; 3],
236    pub _pad2: f32,
237    /// Multiplier on `voxel_frac` (= vol_in_tiles - floor(vol_in_tiles)).
238    /// = (tile_size - 1) / atlas_slot_size per axis. Together with
239    /// `data_offset`, maps `voxel_frac` in [0, 1] to the closed atlas-UV
240    /// range [0.5/atlas_slot, (atlas_slot - 0.5)/atlas_slot] — i.e.
241    /// onto the *centres* of the first and last texels of the slot. With
242    /// Nearest filtering this avoids the tie-break at half-texel
243    /// boundaries that produces visible lines at tile edges.
244    pub data_scale: [f32; 3],
245    pub _pad3: f32,
246    /// Half-texel inset added BEFORE scaling within_tile. = 0.5 /
247    /// atlas_slot_size per axis. See `data_scale`.
248    pub data_offset: [f32; 3],
249    pub _pad4: f32,
250}

252/// Uniform block for the VT pipeline.
253/// Layout must match `VTUniforms` in `virtual_tile.wgsl`.
254///
255/// Slot linearisation (cols-major, then rows, then layers):
256///   col   = slot % atlas_cols
257///   row   = (slot / atlas_cols) % atlas_rows
258///   layer = slot / (atlas_cols * atlas_rows)
259#[repr(C)]
260#[derive(Copy, Clone, Debug, Pod, Zeroable)]
261pub struct VTUniforms {
262    pub atlas_cols: u32,
263    pub atlas_rows: u32,
264    /// UV advance per tile: 1/atlas_cols, 1/atlas_rows, 1/atlas_layers.
265    pub atlas_tile_pitch_x: f32,
266    pub atlas_tile_pitch_y: f32,
267    pub atlas_tile_pitch_z: f32,
268    pub lod_count: u32,
269    pub contrast_min: f32,
270    pub contrast_max: f32,
271    /// Non-zero enables debug LOD tinting (green=LOD0, red=coarsest).
272    pub debug_mode: u32,
273    /// Width of the page table texture (for linear→2D index mapping in the shader).
274    pub page_table_width: u32,
275    /// CPU-computed ideal LOD for this frame (accounts for lod_bias and camera distance).
276    /// The shader starts its page-table walk here rather than at LOD 0.
277    pub target_lod: u32,
278    /// Currently-displayed timepoint. The shader compares this against the
279    /// `t` embedded in each page-table texel; a mismatch is treated as
280    /// non-resident, which keeps the display strictly on this t (no
281    /// blending across timepoints) while letting stale entries from a
282    /// previous t persist harmlessly in the page table.
283    pub desired_t: u32,
284    /// Per-visual opacity multiplier in [0, 1], applied to the final
285    /// (premultiplied) fragment output.
286    pub opacity: f32,
287    // Pad the header to a 16-byte boundary so `lods` stays aligned (12 scalars
288    // + opacity + 3 pad = 16 scalars = 64 bytes; 64 mod 16 = 0 ✓).
289    pub _pad_op0: f32,
290    pub _pad_op1: f32,
291    pub _pad_op2: f32,
292    pub lods: [VTLodInfo; VT_MAX_LODS],
293}

445/// Compute the intersection polygon between a plane and an AABB
446///
447/// Returns vertices and indices for rendering the slice, or None if no intersection.
448/// The vertices are sorted to form a convex polygon in counter-clockwise order.
449pub fn compute_plane_aabb_intersection(
450    plane: &SlicePlane,
451    aabb: &AABB,
452) -> Option<(Vec<TileVertex>, Vec<u16>)> {
453    let plane_pos = Vec3::from(plane.position);
454    let plane_normal = Vec3::from(plane.normal);
455    let min = aabb.min;
456    let max = aabb.max;
457
458    // 12 edges of the AABB
459    let edges = [
460        // Bottom face (y = min)
461        (Vec3::new(min.x, min.y, min.z), Vec3::new(max.x, min.y, min.z)),
462        (Vec3::new(max.x, min.y, min.z), Vec3::new(max.x, min.y, max.z)),
463        (Vec3::new(max.x, min.y, max.z), Vec3::new(min.x, min.y, max.z)),
464        (Vec3::new(min.x, min.y, max.z), Vec3::new(min.x, min.y, min.z)),
465        // Top face (y = max)
466        (Vec3::new(min.x, max.y, min.z), Vec3::new(max.x, max.y, min.z)),
467        (Vec3::new(max.x, max.y, min.z), Vec3::new(max.x, max.y, max.z)),
468        (Vec3::new(max.x, max.y, max.z), Vec3::new(min.x, max.y, max.z)),
469        (Vec3::new(min.x, max.y, max.z), Vec3::new(min.x, max.y, min.z)),
470        // Vertical edges
471        (Vec3::new(min.x, min.y, min.z), Vec3::new(min.x, max.y, min.z)),
472        (Vec3::new(max.x, min.y, min.z), Vec3::new(max.x, max.y, min.z)),
473        (Vec3::new(max.x, min.y, max.z), Vec3::new(max.x, max.y, max.z)),
474        (Vec3::new(min.x, min.y, max.z), Vec3::new(min.x, max.y, max.z)),
475    ];
476
477    let mut intersection_points = Vec::new();
478
479    // Find intersections with each edge
480    for (p0, p1) in edges.iter() {
481        if let Some(t) = line_plane_intersection(*p0, *p1, plane_pos, plane_normal) {
482            if (0.0..=1.0).contains(&t) {
483                let point = p0.lerp(*p1, t);
484                // Calculate 3D texture coordinate
485                let texcoord = Vec3::new(
486                    (point.x - min.x) / (max.x - min.x),
487                    (point.y - min.y) / (max.y - min.y),
488                    (point.z - min.z) / (max.z - min.z),
489                );
490                intersection_points.push((point, texcoord));
491            }
492        }
493    }
494
495    // Need at least 3 points for a polygon
496    if intersection_points.len() < 3 {
497        return None;
498    }
499
500    // Remove ALL duplicate points (not just consecutive ones)
501    // Use epsilon relative to AABB size for scale-independence
502    let aabb_diagonal = (max - min).length();
503    let epsilon = aabb_diagonal * 0.001; // 0.1% of AABB diagonal
504    let mut unique_points = Vec::new();
505    for point in intersection_points {
506        let is_duplicate = unique_points.iter().any(|(p, _): &(Vec3, Vec3)| {
507            p.distance(point.0) < epsilon
508        });
509        if !is_duplicate {
510            unique_points.push(point);
511        }
512    }
513
514    if unique_points.len() < 3 {
515        return None;
516    }
517
518    let intersection_points = unique_points;
519
520    // Sort points to form a convex polygon
521    let sorted_points = sort_points_on_plane(&intersection_points, plane_normal);
522
523    // Convert to vertices
524    let vertices: Vec<TileVertex> = sorted_points
525        .iter()
526        .map(|(pos, tex)| TileVertex {
527            position: [pos.x, pos.y, pos.z],
528            texcoord: [tex.x, tex.y, tex.z],
529        })
530        .collect();
531
532    // Generate indices as a triangle fan (0,1,2, 0,2,3, 0,3,4, ...)
533    let mut indices = Vec::new();
534    for i in 1..(vertices.len() - 1) {
535        indices.push(0);
536        indices.push(i as u16);
537        indices.push((i + 1) as u16);
538    }
539
540    Some((vertices, indices))
541}

543/// Find the intersection point of a line segment with a plane
544///
545/// Returns t value where intersection = p0 + t * (p1 - p0)
546fn line_plane_intersection(p0: Vec3, p1: Vec3, plane_pos: Vec3, plane_normal: Vec3) -> Option<f32> {
547    let ray_dir = p1 - p0;
548    let denom = plane_normal.dot(ray_dir);
549
550    // Line is parallel to plane
551    if denom.abs() < 1e-6 {
552        return None;
553    }
554
555    let t = plane_normal.dot(plane_pos - p0) / denom;
556    Some(t)
557}

559/// Sort points to form a convex polygon in counter-clockwise order
560fn sort_points_on_plane(points: &[(Vec3, Vec3)], plane_normal: Vec3) -> Vec<(Vec3, Vec3)> {
561    if points.is_empty() {
562        return Vec::new();
563    }
564
565    // Find centroid
566    let centroid = points.iter().map(|(p, _)| *p).sum::<Vec3>() / points.len() as f32;
567
568    // Create a coordinate system on the plane
569    let up = if plane_normal.y.abs() < 0.9 {
570        Vec3::Y
571    } else {
572        Vec3::X
573    };
574
575    let tangent_unnormalized = plane_normal.cross(up);
576    let tangent = if tangent_unnormalized.length() > 1e-6 {
577        tangent_unnormalized.normalize()
578    } else {
579        // Fallback if cross product is too small
580        Vec3::X
581    };
582
583    let bitangent_unnormalized = plane_normal.cross(tangent);
584    let bitangent = if bitangent_unnormalized.length() > 1e-6 {
585        bitangent_unnormalized.normalize()
586    } else {
587        Vec3::Y
588    };
589
590    // Sort by angle around centroid
591    let mut sorted = points.to_vec();
592    sorted.sort_by(|(a, _), (b, _)| {
593        let va = *a - centroid;
594        let vb = *b - centroid;
595
596        let angle_a = va.dot(tangent).atan2(va.dot(bitangent));
597        let angle_b = vb.dot(tangent).atan2(vb.dot(bitangent));
598
599        // Handle NaN safely
600        angle_a.partial_cmp(&angle_b).unwrap_or(std::cmp::Ordering::Equal)
601    });
602
603    sorted
604}

9/// Slice plane defined by position and normal vector
10#[derive(Debug, Clone, Copy)]
11pub struct SlicePlane {
12    pub position: [f32; 3],
13    pub normal: [f32; 3],
14}
15
16impl SlicePlane {
17    /// Create a plane perpendicular to Z axis (XY slice)
18    pub fn xy(z: f32) -> Self {
19        Self {
20            position: [0.0, 0.0, z],
21            normal: [0.0, 0.0, 1.0],
22        }
23    }
24
25    /// Create a plane perpendicular to Y axis (XZ slice)
26    pub fn xz(y: f32) -> Self {
27        Self {
28            position: [0.0, y, 0.0],
29            normal: [0.0, 1.0, 0.0],
30        }
31    }
32
33    /// Create a plane perpendicular to X axis (YZ slice)
34    pub fn yz(x: f32) -> Self {
35        Self {
36            position: [x, 0.0, 0.0],
37            normal: [1.0, 0.0, 0.0],
38        }
39    }
40
41    /// Create an arbitrary slice plane with given position and normal
42    /// The normal will be automatically normalized
43    pub fn new(position: [f32; 3], normal: [f32; 3]) -> Self {
44        let len = (normal[0] * normal[0] + normal[1] * normal[1] + normal[2] * normal[2]).sqrt();
45        Self {
46            position,
47            normal: [normal[0] / len, normal[1] / len, normal[2] / len],
48        }
49    }
50}
51
52impl Default for SlicePlane {
53    fn default() -> Self {
54        Self::xy(0.0)
55    }
56}

96    /// Create an Image with virtual-texture multiscale rendering.
97    ///
98    /// # Arguments
99    /// * `lod_levels` - Configuration for each LOD level (high-res to low-res)
100    /// * `max_tiles`  - Total atlas capacity across all atlases
101    /// * `atlas_count` - Number of physical atlas textures (1..=MAX_ATLAS_COUNT)
102    pub fn new(
103        device: &wgpu::Device,
104        queue: &wgpu::Queue,
105        surface_format: wgpu::TextureFormat,
106        camera_bind_group_layout: &wgpu::BindGroupLayout,
107        lod_levels: Vec<LodLevelConfig>,
108        max_tiles: usize,
109        atlas_count: usize,
110    ) -> Self {
111        let (depth, _height, _width) = if !lod_levels.is_empty() {
112            lod_levels[0].volume_size
113        } else {
114            (1, 1, 1)
115        };
116
117        let vt = VirtualTextureData::new(device, lod_levels, max_tiles, atlas_count);
118
119        // Atlas sampler (linear for smooth interpolation across tile boundaries)
120        let sampler = device.create_sampler(&wgpu::SamplerDescriptor {
121            label: Some("VT Sampler"),
122            address_mode_u: wgpu::AddressMode::ClampToEdge,
123            address_mode_v: wgpu::AddressMode::ClampToEdge,
124            address_mode_w: wgpu::AddressMode::ClampToEdge,
125            mag_filter: wgpu::FilterMode::Nearest,
126            min_filter: wgpu::FilterMode::Nearest,
127            mipmap_filter: wgpu::FilterMode::Nearest,
128            ..Default::default()
129        });
130
131        // Colormap LUT bind group layout (group 2)
132        let colormap_bind_group_layout = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
133            label: Some("Colormap Bind Group Layout"),
134            entries: &[
135                wgpu::BindGroupLayoutEntry {
136                    binding: 0,
137                    visibility: wgpu::ShaderStages::FRAGMENT,
138                    ty: wgpu::BindingType::Texture {
139                        sample_type: wgpu::TextureSampleType::Float { filterable: true },
140                        view_dimension: wgpu::TextureViewDimension::D2,
141                        multisampled: false,
142                    },
143                    count: None,
144                },
145                wgpu::BindGroupLayoutEntry {
146                    binding: 1,
147                    visibility: wgpu::ShaderStages::FRAGMENT,
148                    ty: wgpu::BindingType::Sampler(wgpu::SamplerBindingType::Filtering),
149                    count: None,
150                },
151            ],
152        });
153
154        // Build default grayscale colormap (256 RGBA entries)
155        let colormap_data: Vec<u8> = (0u32..256)
156            .flat_map(|i| { let v = i as u8; [v, v, v, 255] })
157            .collect();
158        let colormap_texture = device.create_texture(&wgpu::TextureDescriptor {
159            label: Some("Colormap Texture"),
160            size: wgpu::Extent3d { width: 256, height: 1, depth_or_array_layers: 1 },
161            mip_level_count: 1,
162            sample_count: 1,
163            // 2D (not 1D) — see volume.rs for the naga/GL 1D-Rgba8Unorm bug.
164            dimension: wgpu::TextureDimension::D2,
165            format: wgpu::TextureFormat::Rgba8Unorm,
166            usage: wgpu::TextureUsages::TEXTURE_BINDING | wgpu::TextureUsages::COPY_DST,
167            view_formats: &[],
168        });
169        queue.write_texture(
170            wgpu::TexelCopyTextureInfo {
171                texture: &colormap_texture,
172                mip_level: 0,
173                origin: wgpu::Origin3d::ZERO,
174                aspect: wgpu::TextureAspect::All,
175            },
176            &colormap_data,
177            wgpu::TexelCopyBufferLayout { offset: 0, bytes_per_row: Some(256 * 4), rows_per_image: None },
178            wgpu::Extent3d { width: 256, height: 1, depth_or_array_layers: 1 },
179        );
180        let colormap_sampler = device.create_sampler(&wgpu::SamplerDescriptor {
181            label: Some("Colormap Sampler"),
182            address_mode_u: wgpu::AddressMode::ClampToEdge,
183            mag_filter: wgpu::FilterMode::Linear,
184            min_filter: wgpu::FilterMode::Linear,
185            ..Default::default()
186        });
187        let colormap_view = colormap_texture.create_view(&wgpu::TextureViewDescriptor::default());
188        let colormap_bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
189            label: Some("Colormap Bind Group"),
190            layout: &colormap_bind_group_layout,
191            entries: &[
192                wgpu::BindGroupEntry { binding: 0, resource: wgpu::BindingResource::TextureView(&colormap_view) },
193                wgpu::BindGroupEntry { binding: 1, resource: wgpu::BindingResource::Sampler(&colormap_sampler) },
194            ],
195        });
196
197        // VT bind group layout (group 1): atlas[0..=3], atlas_sampler, page_table, vt_uniforms.
198        // All 4 atlas textures are always bound; entries past atlas_count are
199        // 1×1×1 dummies created inside VirtualTextureData.
200        use crate::visuals::virtual_texture::MAX_ATLAS_COUNT;
201        let atlas_3d_entry = |binding: u32| wgpu::BindGroupLayoutEntry {
202            binding,
203            visibility: wgpu::ShaderStages::FRAGMENT,
204            ty: wgpu::BindingType::Texture {
205                sample_type: wgpu::TextureSampleType::Float { filterable: true },
206                view_dimension: wgpu::TextureViewDimension::D3,
207                multisampled: false,
208            },
209            count: None,
210        };
211        let vt_bgl = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
212            label: Some("VT Bind Group Layout"),
213            entries: &[
214                atlas_3d_entry(0),
215                atlas_3d_entry(1),
216                atlas_3d_entry(2),
217                atlas_3d_entry(3),
218                wgpu::BindGroupLayoutEntry {
219                    binding: 4,
220                    visibility: wgpu::ShaderStages::FRAGMENT,
221                    ty: wgpu::BindingType::Sampler(wgpu::SamplerBindingType::Filtering),
222                    count: None,
223                },
224                wgpu::BindGroupLayoutEntry {
225                    binding: 5,
226                    visibility: wgpu::ShaderStages::FRAGMENT,
227                    ty: wgpu::BindingType::Texture {
228                        sample_type: wgpu::TextureSampleType::Uint,
229                        view_dimension: wgpu::TextureViewDimension::D2Array,
230                        multisampled: false,
231                    },
232                    count: None,
233                },
234                wgpu::BindGroupLayoutEntry {
235                    binding: 6,
236                    visibility: wgpu::ShaderStages::FRAGMENT,
237                    ty: wgpu::BindingType::Buffer {
238                        ty: wgpu::BufferBindingType::Uniform,
239                        has_dynamic_offset: false,
240                        min_binding_size: None,
241                    },
242                    count: None,
243                },
244            ],
245        });
246
247        let vt_uniform_buffer = device.create_buffer(&wgpu::BufferDescriptor {
248            label: Some("VT Uniform Buffer"),
249            size: std::mem::size_of::<VTUniforms>() as u64,
250            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
251            mapped_at_creation: false,
252        });
253
254        assert_eq!(vt.atlas_views.len(), MAX_ATLAS_COUNT);
255        let vt_bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
256            label: Some("VT Bind Group"),
257            layout: &vt_bgl,
258            entries: &[
259                wgpu::BindGroupEntry { binding: 0, resource: wgpu::BindingResource::TextureView(&vt.atlas_views[0]) },
260                wgpu::BindGroupEntry { binding: 1, resource: wgpu::BindingResource::TextureView(&vt.atlas_views[1]) },
261                wgpu::BindGroupEntry { binding: 2, resource: wgpu::BindingResource::TextureView(&vt.atlas_views[2]) },
262                wgpu::BindGroupEntry { binding: 3, resource: wgpu::BindingResource::TextureView(&vt.atlas_views[3]) },
263                wgpu::BindGroupEntry { binding: 4, resource: wgpu::BindingResource::Sampler(&sampler) },
264                wgpu::BindGroupEntry { binding: 5, resource: wgpu::BindingResource::TextureView(&vt.page_table.texture_view) },
265                wgpu::BindGroupEntry { binding: 6, resource: vt_uniform_buffer.as_entire_binding() },
266            ],
267        });
268
269        let vt_shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
270            label: Some("VT Shader"),
271            source: wgpu::ShaderSource::Wgsl(
272                include_str!("../shaders/virtual_tile.wgsl").into(),
273            ),
274        });
275
276        let vt_pipeline_layout = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
277            label: Some("VT Pipeline Layout"),
278            bind_group_layouts: &[camera_bind_group_layout, &vt_bgl, &colormap_bind_group_layout],
279            push_constant_ranges: &[],
280        });
281
282        let vt_render_pipeline = device.create_render_pipeline(&wgpu::RenderPipelineDescriptor {
283            label: Some("VT Render Pipeline"),
284            layout: Some(&vt_pipeline_layout),
285            vertex: wgpu::VertexState {
286                module: &vt_shader,
287                entry_point: Some("vs_main"),
288                buffers: &[TileVertex::desc()],
289                compilation_options: Default::default(),
290            },
291            fragment: Some(wgpu::FragmentState {
292                module: &vt_shader,
293                entry_point: Some("fs_main"),
294                targets: &[Some(wgpu::ColorTargetState {
295                    format: surface_format,
296                    blend: Some(wgpu::BlendState::ALPHA_BLENDING),
297                    write_mask: wgpu::ColorWrites::ALL,
298                })],
299                compilation_options: Default::default(),
300            }),
301            primitive: wgpu::PrimitiveState {
302                topology: wgpu::PrimitiveTopology::TriangleList,
303                strip_index_format: None,
304                front_face: wgpu::FrontFace::Ccw,
305                cull_mode: None,
306                unclipped_depth: false,
307                polygon_mode: wgpu::PolygonMode::Fill,
308                conservative: false,
309            },
310            depth_stencil: Some(wgpu::DepthStencilState {
311                format: wgpu::TextureFormat::Depth24PlusStencil8,
312                depth_write_enabled: true,
313                depth_compare: wgpu::CompareFunction::LessEqual,
314                stencil: wgpu::StencilState::default(),
315                bias: wgpu::DepthBiasState::default(),
316            }),
317            multisample: wgpu::MultisampleState {
318                count: 1,
319                mask: !0,
320                alpha_to_coverage_enabled: false,
321            },
322            multiview: None,
323            cache: None,
324        });
325
326        let slice_plane = SlicePlane::xy((depth as f32) / 2.0);
327
328        Self {
329            vt,
330            colormap_texture,
331            colormap_bind_group,
332            colormap_bind_group_layout,
333            pending_colormap: None,
334            vt_render_pipeline,
335            vt_bind_group,
336            vt_uniform_buffer,
337            vt_vertex_buffer: None,
338            vt_index_buffer: None,
339            vt_index_count: 0,
340            slice_plane,
341            contrast_limits: (0.0, 1.0),
342            debug_mode: false,
343            frame_number: 0,
344            transform: Transform::identity(),
345            visible: true,
346            name: "Image".to_string(),
347        }
348    }

470impl Visual for Image {
471    fn prepare(&mut self, device: &wgpu::Device, queue: &wgpu::Queue, camera_info: &crate::visual::CameraInfo) {
472        // Upload pending colormap if set
473        if let Some(data) = self.pending_colormap.take() {
474            queue.write_texture(
475                wgpu::TexelCopyTextureInfo {
476                    texture: &self.colormap_texture,
477                    mip_level: 0,
478                    origin: wgpu::Origin3d::ZERO,
479                    aspect: wgpu::TextureAspect::All,
480                },
481                &data,
482                wgpu::TexelCopyBufferLayout { offset: 0, bytes_per_row: Some(256 * 4), rows_per_image: None },
483                wgpu::Extent3d { width: 256, height: 1, depth_or_array_layers: 1 },
484            );
485        }
486
487        self.frame_number += 1;
488
489        let _ = device;
490        self.vt.prepare(queue, &self.slice_plane, self.frame_number, camera_info);
491
492        // Update VT uniform buffer.
493        let vt = &self.vt;
494        let mut lods = [VTLodInfo {
495            grid_dims: [1, 1, 1], _pad: 0,
496            tile_scale: [1.0, 1.0, 1.0], _pad2: 0.0,
497            data_scale: [1.0, 1.0, 1.0], _pad3: 0.0,
498            data_offset: [0.0, 0.0, 0.0], _pad4: 0.0,
499        }; VT_MAX_LODS];
500        // Atlas slot size is fixed at LOD 0's tile dimensions.
501        let (atlas_tile_d, atlas_tile_h, atlas_tile_w) = vt.levels[0].tile_size;
502        for (i, level) in vt.levels.iter().take(VT_MAX_LODS).enumerate() {
503            let (gz, gy, gx) = level.grid_size();
504            let (vol_z, vol_y, vol_x) = level.volume_size;
505            let (tile_d, tile_h, tile_w) = level.tile_size;
506            lods[i] = VTLodInfo {
507                grid_dims: [gx, gy, gz],
508                _pad: 0,
509                tile_scale: [
510                    tile_w as f32 / vol_x as f32,
511                    tile_h as f32 / vol_y as f32,
512                    tile_d as f32 / vol_z as f32,
513                ],
514                _pad2: 0.0,
515                data_scale: [
516                    // See volume.rs for the math: (tile - 1) / atlas
517                    // paired with 0.5/atlas data_offset maps voxel_frac
518                    // [0, 1] to the centres of the first and last
519                    // texels in the slot, avoiding Nearest tie-breaks.
520                    (tile_w as f32 - 1.0) / atlas_tile_w as f32,
521                    (tile_h as f32 - 1.0) / atlas_tile_h as f32,
522                    (tile_d as f32 - 1.0) / atlas_tile_d as f32,
523                ],
524                _pad3: 0.0,
525                data_offset: [
526                    0.5 / atlas_tile_w as f32,
527                    0.5 / atlas_tile_h as f32,
528                    0.5 / atlas_tile_d as f32,
529                ],
530                _pad4: 0.0,
531            };
532        }
533        let uniforms = VTUniforms {
534            atlas_cols: vt.atlas_cols,
535            atlas_rows: vt.atlas_rows,
536            atlas_tile_pitch_x: 1.0 / vt.atlas_cols as f32,
537            atlas_tile_pitch_y: 1.0 / vt.atlas_rows as f32,
538            atlas_tile_pitch_z: 1.0 / vt.atlas_layers as f32,
539            lod_count: vt.levels.len().min(VT_MAX_LODS) as u32,
540            contrast_min: self.contrast_limits.0,
541            contrast_max: self.contrast_limits.1,
542            debug_mode: self.debug_mode as u32,
543            page_table_width: vt.page_table.width,
544            target_lod: vt.cached_ideal_lod as u32,
545            desired_t: vt.desired_t(),
546            lods,
547        };
548        queue.write_buffer(&self.vt_uniform_buffer, 0, bytemuck::cast_slice(&[uniforms]));
549
550        // Rebuild slice-plane intersection geometry for the full volume.
551        let lod0 = &vt.levels[0];
552        let vol_min = glam::Vec3::new(
553            lod0.translation.2,
554            lod0.translation.1,
555            lod0.translation.0,
556        );
557        let vol_max = vol_min + glam::Vec3::new(
558            lod0.volume_size.2 as f32 * lod0.voxel_size.2,
559            lod0.volume_size.1 as f32 * lod0.voxel_size.1,
560            lod0.volume_size.0 as f32 * lod0.voxel_size.0,
561        );
562        let vol_aabb = crate::visuals::gpu_structs::AABB::new(vol_min, vol_max);
563        let tile_plane = crate::visuals::gpu_structs::SlicePlane {
564            position: self.slice_plane.position,
565            normal: self.slice_plane.normal,
566        };
567
568        if let Some((vertices, indices)) =
569            crate::visuals::gpu_structs::compute_plane_aabb_intersection(&tile_plane, &vol_aabb)
570        {
571            use wgpu::util::DeviceExt;
572            self.vt_vertex_buffer =
573                Some(device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
574                    label: Some("VT Vertex Buffer"),
575                    contents: bytemuck::cast_slice(&vertices),
576                    usage: wgpu::BufferUsages::VERTEX,
577                }));
578            self.vt_index_buffer =
579                Some(device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
580                    label: Some("VT Index Buffer"),
581                    contents: bytemuck::cast_slice(&indices),
582                    usage: wgpu::BufferUsages::INDEX,
583                }));
584            self.vt_index_count = indices.len() as u32;
585        } else {
586            self.vt_vertex_buffer = None;
587            self.vt_index_buffer = None;
588            self.vt_index_count = 0;
589        }
590    }
591
592    fn render(&self, render_pass: &mut RenderPass) {
593        if self.vt_index_count == 0 {
594            return;
595        }
596        if let (Some(vb), Some(ib)) = (&self.vt_vertex_buffer, &self.vt_index_buffer) {
597            render_pass.set_pipeline(&self.vt_render_pipeline);
598            render_pass.set_bind_group(1, &self.vt_bind_group, &[]);
599            render_pass.set_bind_group(2, &self.colormap_bind_group, &[]);
600            render_pass.set_vertex_buffer(0, vb.slice(..));
601            render_pass.set_index_buffer(ib.slice(..), wgpu::IndexFormat::Uint16);
602            render_pass.draw_indexed(0..self.vt_index_count, 0, 0..1);
603        }
604    }

612    }

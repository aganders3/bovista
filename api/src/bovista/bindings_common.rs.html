18/// Pack a uint16 tile into `TileData`, normalizing the full u16 range to
19/// [0, 1] and storing as R16Float (the atlas format).
20///
21/// Shared by both bindings so the half-float packing — which is
22/// correctness-sensitive — lives in exactly one place.
23pub fn pack_u16_tile(
24    data: impl IntoIterator<Item = u16>,
25    z_shape: u32,
26    y_shape: u32,
27    x_shape: u32,
28) -> TileData {
29    let bytes = data
30        .into_iter()
31        .flat_map(|v| half::f16::from_f32(v as f32 / u16::MAX as f32).to_le_bytes())
32        .collect();
33    TileData { data: bytes, z_shape, y_shape, x_shape, format: wgpu::TextureFormat::R16Float }
34}

36/// Pack a uint8 tile into `TileData`, normalizing the full u8 range to
37/// [0, 1] and storing as R16Float. Lets callers push native 8-bit data
38/// without pre-scaling it to u16 themselves.
39pub fn pack_u8_tile(
40    data: impl IntoIterator<Item = u8>,
41    z_shape: u32,
42    y_shape: u32,
43    x_shape: u32,
44) -> TileData {
45    let bytes = data
46        .into_iter()
47        .flat_map(|v| half::f16::from_f32(v as f32 / u8::MAX as f32).to_le_bytes())
48        .collect();
49    TileData { data: bytes, z_shape, y_shape, x_shape, format: wgpu::TextureFormat::R16Float }
50}

52// Type alias that works for both platforms
53#[cfg(not(target_arch = "wasm32"))]
54pub type VisualRef = Arc<Mutex<dyn Visual>>;
55#[cfg(target_arch = "wasm32")]
56pub type VisualRef = Rc<RefCell<dyn Visual>>;
57
58/// Generic downcast helper for mutable access - Native version
59///
60/// Safely downcasts a VisualRef to a specific Visual type and applies a closure.
61/// This handles the platform-specific synchronization primitives (Arc<Mutex<>> on native).
62#[cfg(not(target_arch = "wasm32"))]
63pub fn with_visual_mut<T, F, R>(visual_ref: &VisualRef, f: F) -> Result<R, String>
64where
65    T: Visual + 'static,
66    F: FnOnce(&mut T) -> R,
67{
68    let mut visual = visual_ref.lock()
69        .map_err(|e| format!("Lock error: {}", e))?;
70    let visual_any: &mut dyn Any = &mut *visual;
71    visual_any.downcast_mut::<T>()
72        .map(f)
73        .ok_or_else(|| format!("Failed to downcast to {}", std::any::type_name::<T>()))
74}

76/// Generic downcast helper for mutable access - WASM version
77///
78/// Safely downcasts a VisualRef to a specific Visual type and applies a closure.
79/// This handles the platform-specific synchronization primitives (Rc<RefCell<>> on WASM).
80#[cfg(target_arch = "wasm32")]
